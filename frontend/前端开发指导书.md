## 0. 设计哲学与全局美学方向

### 0.1 设计关键词

**温暖学术极简 (Warm Academic Minimal)**

不是冰冷的科技黑，不是花哨的渐变紫，而是一种"精心排版的教科书"气质——温和、精确、有节奏感。

### 0.2 三条底线

| 原则 | 说明 |
|------|------|
| **信息优先于装饰** | 每个视觉元素必须承载信息功能，不允许纯装饰性图形 |
| **节奏优于密集** | 用留白建立阅读节奏，而非用边框和卡片堆砌密度 |
| **克制的动效** | 动画只在"需要引导注意力"或"需要反馈操作结果"时出现 |

### 0.3 反模式（绝对禁止）

- 紫色渐变背景
- 毛玻璃 + 暗黑模式的"AI套娃"风格
- 所有页面居中 + 三列等宽卡片的"AI默认模板"
- 每个元素独立动画但无编排关系
- Emoji 作为功能图标（统一用 lucide-react）

**输入框样式**（统一所有表单输入）：


关键点：

- **底线 1.5px**，不是默认 1px，稍微加重
- **focus 态底线变绿**（`focus:border-accent`），无 focus ring
- **无背景色**，透明
- **圆角 0**（`rounded-none`），与全局 2px 圆角解耦

**按钮样式**：


**StatCard 组件**（`components/stat-card.tsx`）：

```tsx
interface StatCardProps {
  label: string
  value: string
  sub?: string
  ring?: { current: number; total: number } // 用于正确率进度环
}

export function StatCard({ label, value, sub, ring }: StatCardProps) {
  return (
    <div className="bg-surface p-6 rounded hover:shadow-card-hover
                    hover:-translate-y-0.5 transition-all duration-200 cursor-default">
      <p className="text-2xs uppercase tracking-widest text-ink-3 mb-2.5">
        {label}
      </p>
      {ring ? (
        <div className="flex items-center gap-3.5">
          {/* SVG 进度环 */}
          <svg className="w-12 h-12 -rotate-90" viewBox="0 0 48 48">
            <circle className="fill-none stroke-surface-3" cx="24" cy="24" r="20" strokeWidth="4" />
            <circle
              className="fill-none stroke-accent transition-[stroke-dashoffset] duration-1000 ease-out"
              cx="24" cy="24" r="20" strokeWidth="4"
              strokeLinecap="round"
              strokeDasharray={`${2 * Math.PI * 20}`}
              strokeDashoffset={`${2 * Math.PI * 20 * (1 - ring.current / ring.total)}`}
            />
          </svg>
          <span className="font-display text-2xl">{value}</span>
        </div>
      ) : (
        <p className="font-display text-2xl">{value}</p>
      )}
      {sub && <p className="text-xs text-ink-3 mt-1.5">{sub}</p>}
    </div>
  )
}
```

**入场动画编排**：

```tsx
<div className="grid grid-cols-4 gap-4 mb-10">
```

每张卡片延迟 80ms 入场，形成从左到右的"波浪"效果。

```tsx
function typeIcon(type: ResourceType) {
```

### 6.7 StreamingText 流式渲染

```tsx
// hooks/use-sse-stream.ts
'use client'
import { useState, useCallback, useRef } from 'react'
import type { SSEEvent } from '@/types'

interface UseSSEStreamOptions {
  url: string
  body: object
  onMessage?: (event: SSEEvent) => void
  onDone?: () => void
  onError?: (error: Error) => void
}

export function useSSEStream() {
  const [content, setContent] = useState('')
  const [isStreaming, setIsStreaming] = useState(false)
  const abortRef = useRef<AbortController | null>(null)

  const start = useCallback(async (url: string, body: object) => {
    abortRef.current = new AbortController()
    setIsStreaming(true)
    setContent('')

    try {
      const res = await fetch(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
        signal: abortRef.current.signal,
      })

      const reader = res.body!.getReader()
      const decoder = new TextDecoder()

      while (true) {
        const { done, value } = await reader.read()
        if (done) break

        const chunk = decoder.decode(value, { stream: true })
        // 解析 SSE 格式
        const lines = chunk.split('\n')
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = JSON.parse(line.slice(6)) as SSEEvent
            if (data.type === 'message' && data.data?.content) {
              setContent(prev => prev + data.data!.content)
            }
          }
        }
      }

      setIsStreaming(false)
    } catch (err) {
      if ((err as Error).name !== 'AbortError') {
        setIsStreaming(false)
      }
    }
  }, [])

  const stop = useCallback(() => {
    abortRef.current?.abort()
    setIsStreaming(false)
  }, [])

  return { content, isStreaming, start, stop, setContent }
}
```

```tsx
        <textarea
          ref={textareaRef}
          value={value}

          onKeyDown={handleKeyDown}
```

```tsx
import { Background, Controls, type Node, type Edge } from 'reactflow'

const nodeTypes = { pathNode: PathNode }

export function PathGraph({ nodes, edges, onNodeClick }: PathGraphProps) {
  const flowNodes: Node[] = nodes.map(n => ({
    id: n.id,
    type: 'pathNode',
    position: { x: n.x, y: n.y },
    data: { label: n.name, status: n.status },
  }))

  const flowEdges: Edge[] = edges.map(([from, to]) => ({
    id: `${from}-${to}`,
    source: from,
    target: to,
    animated: false,
    style: { stroke: '#d8d1c8', strokeWidth: 1.5 },
    type: 'smoothstep',
  }))

  return (
    <div className="bg-surface rounded p-4 h-[400px]">
      <ReactFlow
        nodes={flowNodes}
        edges={flowEdges}
        nodeTypes={nodeTypes}
        onNodeClick={(_, node) => onNodeClick(node.id)}
        fitView
        proOptions={{ hideAttribution: true }}
      >
        <Background color="rgba(44,37,32,0.04)" gap={24} size={1} />
        <Controls
          className="!bg-surface !border-surface-3 !rounded"
          showInteractive={false}
        />
      </ReactFlow>
```

```ts
import { create } from 'zustand'

interface GlobalStore {
  // ── 用户 ──
  user: User | null
  setUser: (user: User | null) => void

  // ── 对话 ──
  currentSessionId: string | null
  messages: Message[]
  isStreaming: boolean
  addMessage: (msg: Message) => void
  appendToLastMessage: (chunk: string) => void
  setStreaming: (v: boolean) => void
  clearMessages: () => void

  // ── Agent 进度 ──
  agentProgress: Record<string, AgentProgress>
  updateAgentProgress: (agent: string, progress: AgentProgress) => void
  resetAgentProgress: () => void

  // ── 画像 ──
  profile: StudentProfile | null
  setProfile: (p: StudentProfile) => void

  // ── 路径 ──
  learningPath: LearningPath | null
  setLearningPath: (p: LearningPath) => void
  selectedNodeId: string | null
  setSelectedNodeId: (id: string | null) => void
}

export const useStore = create<GlobalStore>()((set) => ({
  user: null,
  setUser: (user) => set({ user }),

  currentSessionId: null,
  messages: [],
  isStreaming: false,
  addMessage: (msg) => set((s) => ({ messages: [...s.messages, msg] })),
  appendToLastMessage: (chunk) => set((s) => {
    const msgs = [...s.messages]
    const last = msgs[msgs.length - 1]
    if (last) {
      msgs[msgs.length - 1] = { ...last, content: last.content + chunk }
    }
    return { messages: msgs }
  }),
  setStreaming: (v) => set({ isStreaming: v }),
  clearMessages: () => set({ messages: [], currentSessionId: null }),

  agentProgress: {},
  updateAgentProgress: (agent, progress) => set((s) => ({
    agentProgress: { ...s.agentProgress, [agent]: progress },
  })),
  resetAgentProgress: () => set({ agentProgress: {} }),

  profile: null,
  setProfile: (p) => set({ profile: p }),

  learningPath: null,
  setLearningPath: (p) => set({ learningPath: p }),
  selectedNodeId: null,
  setSelectedNodeId: (id) => set({ selectedNodeId: id }),
}))
```

```ts
// lib/api.ts
const BASE = '/api/v1'

async function request<T>(method: string, path: string, body?: object): Promise<T> {
  const token = localStorage.getItem('zhishu-token')
  const res = await fetch(`${BASE}${path}`, {
    method,
    headers: {
      'Content-Type': 'application/json',
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
    },
    body: body ? JSON.stringify(body) : undefined,
  })

  if (!res.ok) {
    if (res.status === 401) {
      // Token 过期 → 跳转登录
      window.location.href = '/login'
    }
    throw new Error(`API Error: ${res.status}`)
  }
  return res.json()
}

export const api = {
```

### 13.3 SSE 对话流

```ts
// hooks/use-chat-stream.ts
'use client'
import { useCallback } from 'react'
import { useStore } from '@/lib/store'
import type { SSEEvent } from '@/types'

export function useChatStream() {
  const addMessage = useStore(s => s.addMessage)
  const appendToLastMessage = useStore(s => s.appendToLastMessage)
  const setStreaming = useStore(s => s.setStreaming)
  const updateAgentProgress = useStore(s => s.updateAgentProgress)

  const send = useCallback(async (content: string) => {
    // 添加用户消息
    addMessage({ role: 'user', content, id: crypto.randomUUID() })

    // 添加空的 AI 消息（用于流式追加）
    const aiMsgId = crypto.randomUUID()
    addMessage({ role: 'assistant', content: '', id: aiMsgId })

    setStreaming(true)

    try {
      const res = await fetch('/api/v1/chat/stream', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ message: content }),
      })

      const reader = res.body!.getReader()
      const decoder = new TextDecoder()
      let buffer = ''

      while (true) {
        const { done, value } = await reader.read()
        if (done) break

        buffer += decoder.decode(value, { stream: true })
        const lines = buffer.split('\n')
        buffer = lines.pop() || '' // 保留不完整的行

        for (const line of lines) {
          if (!line.startsWith('data: ')) continue
          try {
            const event: SSEEvent = JSON.parse(line.slice(6))

            switch (event.type) {
              case 'message':
                if (event.data?.content) {
                  appendToLastMessage(event.data.content)
                }
                break

              case 'progress':
                if (event.agent) {
                  updateAgentProgress(event.agent, {
                    status: 'active',
                    progress: event.progress || 0,
                    step: event.step,
                  })
                }
                break

              case 'result':
                // Agent 完成
                if (event.agent) {
                  updateAgentProgress(event.agent, {
                    status: 'done',
                    progress: 1,
                  })
                }
                break

              case 'done':
                setStreaming(false)
                break

              case 'error':
                setStreaming(false)
                break
            }
          } catch {}
        }
      }
    } catch (err) {
      setStreaming(false)
    }
  }, [addMessage, appendToLastMessage, setStreaming, updateAgentProgress])

  return { send }
}
```
