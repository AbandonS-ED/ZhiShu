## 1. 项目概述

### 1.1 背景

当前高校《人工智能导论》课程面临以下痛点：

| 痛点 | 描述 |
|------|------|
| 资源单一 | 课件/PPT 内容固化，无法根据学生个体差异动态调整 |
| 缺乏个性化 | 所有学生获得相同学习材料，忽视认知水平差异 |
| 互动性不足 | 缺少即时问答、代码练习反馈等交互手段 |
| 学习路径模糊 | 学生不清楚知识点间的先修关系，盲目学习效率低 |
| 评估滞后 | 仅靠期末考试评估，无法实时追踪学习状态 |

### 1.2 项目目标

构建一个基于多智能体协作的自适应学习资源生成平台，核心能力：

- **个性化画像**：六维学习画像精准刻画学生认知状态
- **智能资源生成**：基于 RAG + LLM 自动生成个性化学习材料
- **自适应路径**：拓扑排序 + 弱项优先生成最优学习路径
- **多模态输出**：文本、思维导图、音频、代码示例一站式生成
- **防幻觉保障**：基于知识库验证的可信内容生成

### 1.3 核心功能清单

| 编号 | 功能 | 描述 |
|------|------|------|
| F1 | 学习者画像构建 | 通过问卷 + 行为分析生成六维画像（知识掌握、学习风格、认知水平、兴趣偏好、薄弱环节、学习节奏） |
| F2 | 个性化资源生成 | 根据画像自动生成：知识点讲解、思维导图、练习题、代码示例、学习脚本音频 |
| F3 | 自适应学习路径 | 基于知识图谱拓扑排序 + 画像弱项优先，生成个性化学习路径 |
| F4 | 智能辅导问答 | RAG 增强的课程问答，支持上下文追问、代码调试 |
| F5 | 学习进度追踪 | 实时记录学习行为，动态更新画像与路径 |

```python
def _build_auth_url(base_url: str, api_key: str, api_secret: str) -> str:
    """构建带 HMAC-SHA256 签名的 WebSocket 认证 URL"""
    parsed = urlparse(base_url)
    host = parsed.hostname
    path = parsed.path
    now = datetime.utcnow()
    date = now.strftime("%a, %d %b %Y %H:%M:%S GMT")

    signature_origin = f"host: {host}\ndate: {date}\nGET {path} HTTP/1.1"
    signature_sha = hmac.new(
        api_secret.encode("utf-8"),
        signature_origin.encode("utf-8"),
        digestmod=hashlib.sha256,
    ).digest()
    signature = base64.b64encode(signature_sha).decode("utf-8")

    authorization_origin = (
        f'api_key="{api_key}", '
        f'algorithm="hmac-sha256", '
        f'headers="host date request-line", '
        f'signature="{signature}"'
    )
    authorization = base64.b64encode(authorization_origin.encode("utf-8")).decode(
        "utf-8"
    )

    params = {"authorization": authorization, "date": date, "host": host}
    return f"{base_url}?{urlencode(params)}"


class SparkClient:
    """讯飞星火 API 客户端"""

    def __init__(self, config: Optional[SparkConfig] = None):
        self.config = config or SparkConfig(
            app_id=settings.SPARK_APP_ID,
            api_key=settings.SPARK_API_KEY,
            api_secret=settings.SPARK_API_SECRET,
        )

    async def _get_http_client(self) -> httpx.AsyncClient:
        if self._http_client is None or self._http_client.is_closed:
            self._http_client = httpx.AsyncClient(timeout=60.0)
        return self._http_client

    async def close(self):
        if self._http_client and not self._http_client.is_closed:
            await self._http_client.aclose()

    # ------------------------------------------------------------------
    # 对话补全 (WebSocket)
    # ------------------------------------------------------------------
    async def chat(
        self,
        messages: list[SparkMessage],
        temperature: float = 0.7,
        max_tokens: int = 4096,
        top_k: int = 4,
    ) -> SparkResponse:
        """同步对话（WebSocket）"""
        try:
            import websockets

            url = _build_auth_url(
                self.config.spark_url,
                self.config.api_key,
                self.config.api_secret,
            )

            request_data = {
                "header": {"app_id": self.config.app_id, "uid": "zhishu_user"},
                "parameter": {
                    "chat": {
                        "domain": self.config.domain,
                        "temperature": temperature,
                        "max_tokens": max_tokens,
                        "top_k": top_k,
                    }
                },
                "payload": {
                    "message": {
                        "text": [m.model_dump() for m in messages]
                    }
                },
            }

            full_content = ""
            usage = SparkUsage()

            async with websockets.connect(url) as ws:
                await ws.send(json.dumps(request_data))

                while True:
                    resp_raw = await ws.recv()
                    resp = json.loads(resp_raw)

                    header = resp.get("header", {})
                    code = header.get("code", 0)
                    if code != 0:
                        raise Exception(
                            f"Spark API error: code={code}, "
                            f"message={header.get('message', 'Unknown')}"
                        )

                    payload = resp.get("payload", {})
                    choices = payload.get("choices", {})
                    text_list = choices.get("text", [])

                    for item in text_list:
                        full_content += item.get("content", "")

                    usage_info = payload.get("usage", {})
                    if usage_info:
                        usage.prompt_tokens = usage_info.get("text", {}).get(
                            "prompt_tokens", 0
                        )
                        usage.completion_tokens = usage_info.get("text", {}).get(
                            "completion_tokens", 0
                        )
                        usage.total_tokens = (
                            usage.prompt_tokens + usage.completion_tokens
                        )

                    if header.get("status") == 2:
                        break

            return SparkResponse(content=full_content, usage=usage)

        except Exception as e:
            logger.error(f"Spark chat error: {e}")
            raise

    # Embedding (HTTP REST)
    # ------------------------------------------------------------------

        embedding = embedding_data.get("embedding", [])
        token_count = data.get("usage", {}).get("prompt_tokens", 0)
```

```python
def _convert_message(msg: BaseMessage) -> SparkMessage:
    """LangChain Message -> SparkMessage"""
    if isinstance(msg, SystemMessage):
        return SparkMessage(role="system", content=msg.content)
    elif isinstance(msg, HumanMessage):
        return SparkMessage(role="user", content=msg.content)

        run_manager: Optional[Any] = None,
        **kwargs: Any,
    ) -> ChatResult:
        import asyncio

        spark_msgs = [_convert_message(m) for m in messages]

        loop = asyncio.get_event_loop()
        if loop.is_running():
            import concurrent.futures

            with concurrent.futures.ThreadPoolExecutor() as pool:
                future = pool.submit(
                    asyncio.run,
                    spark_client.chat(
                        spark_msgs,
                        temperature=self.temperature,
                        max_tokens=self.max_tokens,
                    ),
                )
                response = future.result()
        else:
            response = asyncio.run(
                spark_client.chat(
                    spark_msgs,
                    temperature=self.temperature,
                    max_tokens=self.max_tokens,
                )
            )

        message = AIMessage(content=response.content)
        generation = ChatGeneration(message=message)
        return ChatResult(generations=[generation])

    def _stream(
        self,
        messages: list[BaseMessage],
        stop: Optional[list[str]] = None,
        run_manager: Optional[Any] = None,
        **kwargs: Any,
    ) -> Iterator[AIMessageChunk]:
        import asyncio

        spark_msgs = [_convert_message(m) for m in messages]

        async def _stream_gen():
            async for chunk in spark_client.chat_stream(
                spark_msgs,
                temperature=self.temperature,
                max_tokens=self.max_tokens,
            ):
                yield chunk

        loop = asyncio.new_event_loop()
        try:
            gen = _stream_gen()
            while True:
                try:
                    chunk = loop.run_until_complete(gen.__anext__())
                    yield AIMessageChunk(content=chunk)
                except StopAsyncIteration:
                    break
        finally:
            loop.close()

    @property
    def _identifying_params(self) -> dict:
        return {
            "model_name": self.model_name,
            "temperature": self.temperature,
            "max_tokens": self.max_tokens,
        }


class SparkEmbeddings(Embeddings):
    """LangChain 兼容的讯飞星火 Embeddings"""

    def embed_documents(self, texts: list[str]) -> list[list[float]]:
        import asyncio

        results = []
        for text in texts:
            resp = asyncio.run(spark_client.embed(text))
            results.append(resp.embedding)
        return results

    def embed_query(self, text: str) -> list[float]:
        import asyncio

        resp = asyncio.run(spark_client.embed(text))
        return resp.embedding
```

```python
# backend/app/services/cache_schema.py
"""
Redis 缓存数据结构设计
使用不同的 key prefix 区分业务场景
"""

from typing import Optional
import json
import redis.asyncio as redis
from app.core.config import settings

# Redis 连接池
redis_pool: Optional[redis.Redis] = None


async def get_redis() -> redis.Redis:
    global redis_pool
    if redis_pool is None:
        redis_pool = redis.from_url(
            settings.REDIS_URL,
            encoding="utf-8",
            decode_responses=True,
        )
    return redis_pool


# ============================================================
# Key 设计规范:
#   zhishu:{namespace}:{entity}:{id}[:{sub}]
# ============================================================

class CacheKeys:
    """Redis Key 常量定义"""

    # --- 学生会话 ---
    # Hash: 存储当前活跃会话信息
    # Field: session_id, last_active, agent_state
    SESSION = "zhishu:session:{student_id}"

    # --- 学生画像缓存 ---
    # String (JSON): 画像快照, TTL=1h
    PROFILE = "zhishu:profile:{student_id}:{course_id}"

    # --- 对话上下文 ---
    # List: 最近 N 条对话消息 (JSON)
    CONVERSATION = "zhishu:conv:{session_id}"

    # --- 生成任务进度 ---
    # Hash: task_id, status, progress, current_step, message
    TASK_PROGRESS = "zhishu:task:{task_id}"

    # --- 任务进度 Pub/Sub 频道 ---
    TASK_CHANNEL = "zhishu:task_channel:{task_id}"

    # --- Agent 状态 ---
    # String (JSON): Agent 当前执行状态
    AGENT_STATE = "zhishu:agent:{agent_name}:{task_id}"

    # --- RAG 查询缓存 ---
    # String (JSON): 相似查询的检索结果缓存, TTL=30min
    RAG_CACHE = "zhishu:rag_cache:{query_hash}"

    # --- 学习路径缓存 ---
    # String (JSON): 生成的学习路径, TTL=2h
    PATH_CACHE = "zhishu:path:{student_id}:{course_id}"

    # --- 热门知识点 ---
    # Sorted Set: 知识点访问频率统计
    KP_HOTNESS = "zhishu:kp_hotness:{course_id}"

    # --- Rate Limiting ---
    # String: 请求计数器, TTL=1min
    RATE_LIMIT = "zhishu:ratelimit:{student_id}:{endpoint}"


class CacheService:
    """Redis 缓存服务"""

    def __init__(self):
        self._redis: Optional[redis.Redis] = None

    async def init(self):
        self._redis = await get_redis()

    @property
    def r(self) -> redis.Redis:
        assert self._redis is not None, "Redis not initialized"
        return self._redis

    # --- 画像缓存 ---
    async def get_profile(self, student_id: str, course_id: str) -> Optional[dict]:
        key = CacheKeys.PROFILE.format(student_id=student_id, course_id=course_id)
        data = await self.r.get(key)
        return json.loads(data) if data else None

    async def set_profile(
        self, student_id: str, course_id: str, profile: dict, ttl: int = 3600
    ):
        key = CacheKeys.PROFILE.format(student_id=student_id, course_id=course_id)
        await self.r.set(key, json.dumps(profile, ensure_ascii=False), ex=ttl)

    async def invalidate_profile(self, student_id: str, course_id: str):
        key = CacheKeys.PROFILE.format(student_id=student_id, course_id=course_id)
        await self.r.delete(key)

    # --- 对话上下文 ---
    async def add_conversation(self, session_id: str, message: dict, max_len: int = 50):
        key = CacheKeys.CONVERSATION.format(session_id=session_id)
        await self.r.lpush(key, json.dumps(message, ensure_ascii=False))
        await self.r.ltrim(key, 0, max_len - 1)
        await self.r.expire(key, 7200)  # 2h TTL

    async def get_conversation(self, session_id: str, limit: int = 20) -> list[dict]:
        key = CacheKeys.CONVERSATION.format(session_id=session_id)
        data = await self.r.lrange(key, 0, limit - 1)
        return [json.loads(d) for d in data]

    # --- 任务进度 ---
    async def update_task_progress(
        self,
        task_id: str,
        status: str,
        progress: float,
        current_step: str = "",
        message: str = "",
    ):
        key = CacheKeys.TASK_PROGRESS.format(task_id=task_id)
        data = {
            "task_id": task_id,
            "status": status,
            "progress": progress,
            "current_step": current_step,
            "message": message,
        }
        await self.r.hset(key, mapping=data)
        await self.r.expire(key, 3600)

        # 发布进度更新到频道 (供 SSE/WS 推送)
        channel = CacheKeys.TASK_CHANNEL.format(task_id=task_id)
        await self.r.publish(channel, json.dumps(data, ensure_ascii=False))

    async def get_task_progress(self, task_id: str) -> Optional[dict]:
        key = CacheKeys.TASK_PROGRESS.format(task_id=task_id)
        data = await self.r.hgetall(key)
        return data if data else None

    # --- RAG 查询缓存 ---
    async def get_rag_cache(self, query_hash: str) -> Optional[list]:
        key = CacheKeys.RAG_CACHE.format(query_hash=query_hash)
        data = await self.r.get(key)
        return json.loads(data) if data else None

    async def set_rag_cache(self, query_hash: str, results: list, ttl: int = 1800):
        key = CacheKeys.RAG_CACHE.format(query_hash=query_hash)
        await self.r.set(key, json.dumps(results, ensure_ascii=False), ex=ttl)

    # --- 学习路径缓存 ---
    async def get_path(self, student_id: str, course_id: str) -> Optional[dict]:
        key = CacheKeys.PATH_CACHE.format(student_id=student_id, course_id=course_id)
        data = await self.r.get(key)
        return json.loads(data) if data else None

    async def set_path(
        self, student_id: str, course_id: str, path: dict, ttl: int = 7200
    ):
        key = CacheKeys.PATH_CACHE.format(student_id=student_id, course_id=course_id)
        await self.r.set(key, json.dumps(path, ensure_ascii=False), ex=ttl)

    # --- 知识点热度 ---
    async def record_kp_access(self, course_id: str, kp_id: str):
        key = CacheKeys.KP_HOTNESS.format(course_id=course_id)
        await self.r.zincrby(key, 1, kp_id)

    async def get_hot_kps(self, course_id: str, top_n: int = 10) -> list[str]:
        key = CacheKeys.KP_HOTNESS.format(course_id=course_id)
        return await self.r.zrevrange(key, 0, top_n - 1)

    # --- Rate Limiting ---
    async def check_rate_limit(
        self, student_id: str, endpoint: str, max_requests: int = 60
    ) -> bool:
        key = CacheKeys.RATE_LIMIT.format(student_id=student_id, endpoint=endpoint)
        count = await self.r.incr(key)
        if count == 1:
            await self.r.expire(key, 60)
        return count <= max_requests

    # --- Pub/Sub ---
    async def subscribe_task(self, task_id: str):
        channel = CacheKeys.TASK_CHANNEL.format(task_id=task_id)
        pubsub = self.r.pubsub()
        await pubsub.subscribe(channel)
        return pubsub


# 全局实例
cache_service = CacheService()
```

```python
# backend/app/services/document_parser.py
"""
多格式文档解析器
支持: PDF / DOCX / PPTX / MD / TXT
输出: 分页文本列表 [(page_num, text, metadata), ...]
"""

import hashlib
from pathlib import Path
from dataclasses import dataclass, field

import chardet
from loguru import logger


@dataclass
class PageContent:
    page_number: int
    text: str
    metadata: dict = field(default_factory=dict)


@dataclass
class ParsedDocument:
    file_path: str
    file_type: str
    total_pages: int
    pages: list[PageContent]
    file_hash: str
    metadata: dict = field(default_factory=dict)


def _compute_file_hash(file_path: str) -> str:
    sha256 = hashlib.sha256()
    with open(file_path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            sha256.update(chunk)
    return sha256.hexdigest()


class DocumentParser:
    """统一文档解析入口"""

    SUPPORTED_TYPES = {".pdf", ".docx", ".pptx", ".md", ".txt"}

    def parse(self, file_path: str) -> ParsedDocument:
        path = Path(file_path)
        if not path.exists():
            raise FileNotFoundError(f"文件不存在: {file_path}")

        suffix = path.suffix.lower()
        if suffix not in self.SUPPORTED_TYPES:
            raise ValueError(f"不支持的文件类型: {suffix}")

        parser_map = {
            ".pdf": self._parse_pdf,
            ".docx": self._parse_docx,
            ".pptx": self._parse_pptx,
            ".md": self._parse_text,
            ".txt": self._parse_text,
        }

        pages = parser_map[suffix](file_path)
        file_hash = _compute_file_hash(file_path)

        return ParsedDocument(
            file_path=file_path,
            file_type=suffix,
            total_pages=len(pages),
            pages=pages,
            file_hash=file_hash,
            metadata={"filename": path.name, "size_bytes": path.stat().st_size},
        )

    def _parse_pdf(self, file_path: str) -> list[PageContent]:
        import fitz  # PyMuPDF

        pages = []
        doc = fitz.open(file_path)
        for i, page in enumerate(doc):
            text = page.get_text("text")
            if text.strip():
                pages.append(
                    PageContent(
                        page_number=i + 1,
                        text=text.strip(),
                        metadata={"source": "pdf", "page": i + 1},
                    )
                )
        doc.close()
        return pages

    def _parse_docx(self, file_path: str) -> list[PageContent]:
        from docx import Document

        doc = Document(file_path)
        pages = []
        current_text = []
        page_num = 1

        for para in doc.paragraphs:
            text = para.text.strip()
            if not text:
                continue

            # 按标题分页 (h1 视为新页)
            if para.style.name.startswith("Heading 1") and current_text:
                pages.append(
                    PageContent(
                        page_number=page_num,
                        text="\n".join(current_text),
                        metadata={"source": "docx"},
                    )
                )
                current_text = []
                page_num += 1

            current_text.append(text)

        if current_text:
            pages.append(
                PageContent(
                    page_number=page_num,
                    text="\n".join(current_text),
                    metadata={"source": "docx"},
                )
            )

        return pages

    def _parse_pptx(self, file_path: str) -> list[PageContent]:
        from pptx import Presentation

        prs = Presentation(file_path)
        pages = []

        for i, slide in enumerate(prs.slides):
            texts = []
            for shape in slide.shapes:
                if hasattr(shape, "text") and shape.text.strip():
                    texts.append(shape.text.strip())

            if texts:
                title = ""
                if slide.shapes.title:
                    title = slide.shapes.title.text.strip()

                pages.append(
                    PageContent(
                        page_number=i + 1,
                        text="\n".join(texts),
                        metadata={"source": "pptx", "slide_title": title},
                    )
                )

        return pages

    def _parse_text(self, file_path: str) -> list[PageContent]:
        path = Path(file_path)

        # 自动检测编码
        with open(file_path, "rb") as f:
            raw = f.read()
            detected = chardet.detect(raw)
            encoding = detected.get("encoding", "utf-8")

        with open(file_path, "r", encoding=encoding, errors="ignore") as f:
            content = f.read()

        # Markdown / TXT 按二级标题分页
        if path.suffix.lower() == ".md":
            sections = content.split("\n## ")
            pages = []
            for i, section in enumerate(sections):
                text = section.strip()
                if text:
                    if i > 0:
                        text = "## " + text
                    pages.append(
                        PageContent(
                            page_number=i + 1,
                            text=text,
                            metadata={"source": "markdown"},
                        )
                    )
            return pages if pages else [PageContent(page_number=1, text=content, metadata={"source": "txt"})]
        else:
            # TXT 按固定行数分页
            lines = content.split("\n")
            page_size = 50
            pages = []
            for i in range(0, len(lines), page_size):
                chunk = "\n".join(lines[i : i + page_size])
                if chunk.strip():
                    pages.append(
                        PageContent(
                            page_number=i // page_size + 1,
                            text=chunk.strip(),
                            metadata={"source": "txt"},
                        )
                    )
            return pages
```

### 5.2 文本切片器 (语义切片)

```python
# backend/app/services/text_chunker.py
"""
语义切片器
策略: 按标题层级切分 + 重叠窗口 + token 限制
"""

import re
from dataclasses import dataclass, field

import tiktoken
from loguru import logger


@dataclass
class TextChunk:
    content: str
    chunk_index: int
    section_title: str = ""
    token_count: int = 0
    metadata: dict = field(default_factory=dict)


class SemanticChunker:
    """基于语义的文本切片器"""

    def __init__(
        self,
        max_tokens: int = 512,
        overlap_tokens: int = 64,
        min_chunk_tokens: int = 50,
    ):
        self.max_tokens = max_tokens
        self.overlap_tokens = overlap_tokens
        self.min_chunk_tokens = min_chunk_tokens
        self.encoder = tiktoken.get_encoding("cl100k_base")

    def count_tokens(self, text: str) -> int:
        return len(self.encoder.encode(text))

    def chunk_document(self, pages: list, metadata: dict = None) -> list[TextChunk]:
        """对整个文档进行语义切片"""
        all_chunks = []
        chunk_index = 0

        for page in pages:
            page_chunks = self._chunk_page(
                page.text,
                page.page_number,
                page.metadata.get("section_title", ""),
            )
            for chunk in page_chunks:
                chunk.chunk_index = chunk_index
                chunk.metadata.update(metadata or {})
                chunk.metadata["page_number"] = page.page_number
                all_chunks.append(chunk)
                chunk_index += 1

        return all_chunks

    def _chunk_page(
        self, text: str, page_number: int, default_title: str = ""
    ) -> list[TextChunk]:
        """对单页文本进行语义切片"""
        # 第一步: 按标题分割
        sections = self._split_by_headers(text)

        chunks = []
        for section_title, section_text in sections:
            if not section_text.strip():
                continue

            # 第二步: 按段落分割
            paragraphs = self._split_paragraphs(section_text)

            # 第三步: 合并段落到 chunk (带 overlap)
            section_chunks = self._merge_paragraphs(
                paragraphs, section_title or default_title
            )
            chunks.extend(section_chunks)

        return chunks

    def _split_by_headers(self, text: str) -> list[tuple[str, str]]:
        """按标题分割文本"""
        # 匹配 Markdown 标题 (# ## ### ####)
        header_pattern = re.compile(r"^(#{1,4})\s+(.+)$", re.MULTILINE)

        sections = []
        matches = list(header_pattern.finditer(text))

        if not matches:
            return [("", text)]

        # 标题前的内容
        if matches[0].start() > 0:
            pre_text = text[: matches[0].start()].strip()
            if pre_text:
                sections.append(("", pre_text))

        for i, match in enumerate(matches):
            title = match.group(2).strip()
            start = match.end()
            end = matches[i + 1].start() if i + 1 < len(matches) else len(text)
            content = text[start:end].strip()
            if content:
                sections.append((title, content))

        return sections

    def _split_paragraphs(self, text: str) -> list[str]:
        """按段落分割 (双换行)"""
        paragraphs = re.split(r"\n\s*\n", text)
        return [p.strip() for p in paragraphs if p.strip()]

    def _merge_paragraphs(
        self, paragraphs: list[str], section_title: str
    ) -> list[TextChunk]:
        """合并段落到 chunk, 带 overlap"""
        chunks = []
        current_parts: list[str] = []
        current_tokens = 0

        for para in paragraphs:
            para_tokens = self.count_tokens(para)

            # 单段落超过 max_tokens, 需要拆分
            if para_tokens > self.max_tokens:
                if current_parts:
                    chunk_text = "\n\n".join(current_parts)
                    chunks.append(
                        TextChunk(
                            content=chunk_text,
                            chunk_index=0,
                            section_title=section_title,
                            token_count=current_tokens,
                        )
                    )
                    current_parts = []
                    current_tokens = 0

                # 按句子拆分长段落
                sub_chunks = self._split_long_paragraph(para, section_title)
                chunks.extend(sub_chunks)
                continue

            # 正常合并
            if current_tokens + para_tokens > self.max_tokens and current_parts:
                chunk_text = "\n\n".join(current_parts)
                chunks.append(
                    TextChunk(
                        content=chunk_text,
                        chunk_index=0,
                        section_title=section_title,
                        token_count=current_tokens,
                    )
                )

                # 保留 overlap: 取最后一部分
                overlap_parts = self._get_overlap_parts(current_parts)
                current_parts = overlap_parts
                current_tokens = sum(self.count_tokens(p) for p in overlap_parts)

            current_parts.append(para)
            current_tokens += para_tokens

        # 最后一个 chunk
        if current_parts and current_tokens >= self.min_chunk_tokens:
            chunk_text = "\n\n".join(current_parts)
            chunks.append(
                TextChunk(
                    content=chunk_text,
                    chunk_index=0,
                    section_title=section_title,
                    token_count=current_tokens,
                )
            )

        return chunks

    def _split_long_paragraph(self, text: str, section_title: str) -> list[TextChunk]:
        """拆分超长段落 (按句子)"""
        sentences = re.split(r"(?<=[。！？.!?])", text)
        chunks = []
        current_parts: list[str] = []
        current_tokens = 0

        for sent in sentences:
            sent = sent.strip()
            if not sent:
                continue

            sent_tokens = self.count_tokens(sent)

            if current_tokens + sent_tokens > self.max_tokens and current_parts:
                chunk_text = "".join(current_parts)
                chunks.append(
                    TextChunk(
                        content=chunk_text,
                        chunk_index=0,
                        section_title=section_title,
                        token_count=current_tokens,
                    )
                )
                current_parts = []
                current_tokens = 0

            current_parts.append(sent)
            current_tokens += sent_tokens

        if current_parts:
            chunk_text = "".join(current_parts)
            chunks.append(
                TextChunk(
                    content=chunk_text,
                    chunk_index=0,
                    section_title=section_title,
                    token_count=current_tokens,
                )
            )

        return chunks

    def _get_overlap_parts(self, parts: list[str]) -> list[str]:
        """获取 overlap 部分"""
        overlap_texts: list[str] = []
        token_count = 0

        for part in reversed(parts):
            part_tokens = self.count_tokens(part)
            if token_count + part_tokens > self.overlap_tokens:
                break
            overlap_texts.insert(0, part)
            token_count += part_tokens

        return overlap_texts
```

```python
"""
Embedding 服务
基于讯飞星火 Embedding API
支持批量处理和缓存
"""

import hashlib
from typing import Optional

from loguru import logger

from app.services.spark_client import spark_client
from app.services.cache_service import cache_service


class EmbeddingService:
    """文本向量化服务"""

    def __init__(self):
        self._cache: dict[str, list[float]] = {}

    def _text_hash(self, text: str) -> str:
        return hashlib.md5(text.encode("utf-8")).hexdigest()

    async def embed(self, text: str) -> list[float]:
        """单条文本 Embedding (带内存缓存)"""
        text_hash = self._text_hash(text)

        # 检查内存缓存
        if text_hash in self._cache:
            return self._cache[text_hash]

        # 调用 API
        resp = await spark_client.embed(text)
        embedding = resp.embedding

        # 存入缓存
        self._cache[text_hash] = embedding

        return embedding

    async def embed_batch(self, texts: list[str]) -> list[list[float]]:
        """批量 Embedding"""
        results = []
        for text in texts:
            embedding = await self.embed(text)
            results.append(embedding)
        return results

    async def embed_with_cache(self, text: str) -> list[float]:
        """带 Redis 缓存的 Embedding"""
        text_hash = self._text_hash(text)

        # 检查 Redis 缓存
        cached = await cache_service.get_rag_cache(f"emb:{text_hash}")
        if cached:
            return cached

        # 调用 API
        embedding = await self.embed(text)

        # 存入 Redis (24h TTL)
        await cache_service.set_rag_cache(f"emb:{text_hash}", embedding, ttl=86400)

        return embedding
```

### 5.4 RAG 服务 (检索 + 重排 + 引用)

```python
# backend/app/services/rag_service.py
"""
RAG (Retrieval-Augmented Generation) 服务
完整流程: Query -> Embed -> 向量检索 -> LLM 重排 -> 引用标注
"""

import hashlib
from dataclasses import dataclass, field

from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession
from loguru import logger

from app.services.embedding_service import embedding_service
from app.services.spark_client import spark_client, SparkMessage
from app.services.cache_service import cache_service


@dataclass
class RetrievedChunk:
    chunk_id: str
    content: str
    source_file: str
    page_number: int
    section_title: str
    similarity: float
    rerank_score: float = 0.0
    citation: str = ""


@dataclass
class RAGResult:
    answer: str
    sources: list[RetrievedChunk]
    citations: list[str]
    confidence: float


class RAGService:
    """RAG 检索增强生成服务"""

    def __init__(self):
        self.top_k = 10          # 向量检索返回数
        self.rerank_top_n = 5    # 重排后保留数
        self.similarity_threshold = 0.65  # 相似度阈值

    async def retrieve(
        self,
        query: str,
        course_id: str,
        db: AsyncSession,
        top_k: int = None,
    ) -> list[RetrievedChunk]:
        """向量检索: 查询最相关的文档切片"""
        top_k = top_k or self.top_k

        # 1. 查询向量化
        query_embedding = await embedding_service.embed(query)
        embedding_str = "[" + ",".join(str(v) for v in query_embedding) + "]"

        # 2. pgvector 余弦相似度检索
        sql = text("""
            SELECT
                id::text AS chunk_id,
                content,
                source_file,
                page_number,
                section_title,
                1 - (embedding <=> :embedding::vector) AS similarity
            FROM document_chunks
            WHERE course_id = :course_id
              AND 1 - (embedding <=> :embedding::vector) > :threshold
            ORDER BY embedding <=> :embedding::vector
            LIMIT :top_k
        """)

        result = await db.execute(
            sql,
            {
                "embedding": embedding_str,
                "course_id": course_id,
                "threshold": self.similarity_threshold,
                "top_k": top_k,
            },
        )

        chunks = []
        for row in result.mappings():
            chunks.append(
                RetrievedChunk(
                    chunk_id=row["chunk_id"],
                    content=row["content"],
                    source_file=row["source_file"] or "",
                    page_number=row["page_number"] or 0,
                    section_title=row["section_title"] or "",
                    similarity=float(row["similarity"]),
                )
            )

        return chunks

    async def rerank(
        self, query: str, chunks: list[RetrievedChunk]
    ) -> list[RetrievedChunk]:
        """LLM 重排: 对检索结果进行相关性重排序"""
        if not chunks:
            return []

        # 构建重排 prompt
        chunk_descriptions = []
        for i, chunk in enumerate(chunks):
            chunk_descriptions.append(
                f"[{i+1}] 来源: {chunk.source_file} p.{chunk.page_number}\n"
                f"内容: {chunk.content[:200]}..."
            )

        rerank_prompt = (
            f"用户问题: {query}\n\n"
            f"以下是检索到的文档片段，请按与问题的相关性从高到低排序，"
            f"只返回排序后的编号列表 (如: [3,1,5,2,4]):\n\n"
            + "\n\n".join(chunk_descriptions)
        )

        messages = [
            SparkMessage(role="system", content="你是一个文档检索重排助手。只返回编号列表。"),
            SparkMessage(role="user", content=rerank_prompt),
        ]

        response = await spark_client.chat(messages, temperature=0.1, max_tokens=200)

        # 解析排序结果
        try:
            import re

            numbers = re.findall(r"\d+", response.content)
            ranked_indices = [int(n) - 1 for n in numbers if 0 <= int(n) - 1 < len(chunks)]

            # 按新排序排列, 并添加重排分数
            reranked = []
            for rank, idx in enumerate(ranked_indices[: self.rerank_top_n]):
                chunk = chunks[idx]
                chunk.rerank_score = 1.0 - (rank / len(ranked_indices))
                reranked.append(chunk)

            return reranked

        except Exception as e:
            logger.warning(f"Rerank 解析失败, 使用原始排序: {e}")
            # 回退: 按原始相似度排序
            for chunk in chunks[: self.rerank_top_n]:
                chunk.rerank_score = chunk.similarity
            return chunks[: self.rerank_top_n]

    def generate_citations(self, chunks: list[RetrievedChunk]) -> list[str]:
        """生成引用标注"""
        citations = []
        for i, chunk in enumerate(chunks):
            citation = (
                f"[{i+1}] {chunk.source_file}"
                f"{', 第{chunk.page_number}页' if chunk.page_number else ''}"
                f"{', ' + chunk.section_title if chunk.section_title else ''}"
                f" (相关度: {chunk.similarity:.2f})"
            )
            citations.append(citation)
        return citations

    async def query(
        self,
        question: str,
        course_id: str,
        db: AsyncSession,
        student_context: str = "",
    ) -> RAGResult:
        """完整 RAG 流程: 检索 -> 重排 -> 生成 -> 引用"""
        # 1. 向量检索
        chunks = await self.retrieve(question, course_id, db)

        if not chunks:
            return RAGResult(
                answer="抱歉，知识库中没有找到相关信息。请尝试换个方式提问。",
                sources=[],
                citations=[],
                confidence=0.0,
            )

        # 2. LLM 重排
        reranked_chunks = await self.rerank(question, chunks)

        # 3. 构建上下文
        context_parts = []
        for i, chunk in enumerate(reranked_chunks):
            context_parts.append(
                f"[来源{i+1}] {chunk.source_file} p.{chunk.page_number}\n{chunk.content}"
            )
        context = "\n\n".join(context_parts)

        # 4. 生成回答 (带引用指令)
        system_prompt = (
            "你是一个专业的课程辅导助手。请严格基于提供的参考资料回答问题。\n"
            "要求:\n"
            "1. 只使用参考资料中的信息回答，不要编造内容\n"
            "2. 在回答中用 [来源N] 标注引用\n"
            "3. 如果参考资料不足以回答问题，明确说明\n"
            "4. 使用 Markdown 格式\n"
        )

        if student_context:
            system_prompt += f"\n学生背景: {student_context}\n"

        messages = [
            SparkMessage(role="system", content=system_prompt),
            SparkMessage(
                role="user",
                content=f"参考资料:\n{context}\n\n问题: {question}",
            ),
        ]

        response = await spark_client.chat(messages, temperature=0.3, max_tokens=2048)

        # 5. 生成引用列表
        citations = self.generate_citations(reranked_chunks)

        # 6. 计算置信度
        avg_similarity = (
            sum(c.similarity for c in reranked_chunks) / len(reranked_chunks)
            if reranked_chunks
            else 0.0
        )

        return RAGResult(
            answer=response.content,
            sources=reranked_chunks,
            citations=citations,
            confidence=avg_similarity,
        )

    async def query_stream(
        self,
        question: str,
        course_id: str,
        db: AsyncSession,
        student_context: str = "",
    ):
        """流式 RAG: 检索+重排后, 流式生成"""
        import asyncio

        # 1. 检索 + 重排
        chunks = await self.retrieve(question, course_id, db)
        if not chunks:
            yield {
                "type": "error",
                "content": "知识库中没有找到相关信息。",
            }
            return

        reranked_chunks = await self.rerank(question, chunks)
        citations = self.generate_citations(reranked_chunks)

        # 发送来源信息
        yield {
            "type": "sources",
            "content": [
                {
                    "file": c.source_file,
                    "page": c.page_number,
                    "section": c.section_title,
                    "similarity": c.similarity,
                }
                for c in reranked_chunks
            ],
        }

        # 2. 构建上下文
        context_parts = []
        for i, chunk in enumerate(reranked_chunks):
            context_parts.append(
                f"[来源{i+1}] {chunk.source_file} p.{chunk.page_number}\n{chunk.content}"
            )
        context = "\n\n".join(context_parts)

        system_prompt = (
            "你是一个专业的课程辅导助手。请严格基于提供的参考资料回答问题。\n"
            "在回答中用 [来源N] 标注引用。\n"
        )

        messages = [
            SparkMessage(role="system", content=system_prompt),
            SparkMessage(
                role="user",
                content=f"参考资料:\n{context}\n\n问题: {question}",
            ),
        ]

        # 3. 流式生成
        async for chunk in spark_client.chat_stream(messages, temperature=0.3):
            yield {"type": "content", "content": chunk}

        # 4. 发送引用
        yield {"type": "citations", "content": citations}
        yield {"type": "done"}


# 全局实例
rag_service = RAGService()
```

```python
import asyncio

async def process_document(
    file_path: str,
    course_id: str,
    session_factory,
):
    """处理单个文档: 解析 -> 切片 -> 向量化 -> 存储"""
    logger.info(f"开始处理文档: {file_path}")

    # 1. 解析文档

    logger.info(f"  解析完成: {doc.total_pages} 页, hash={doc.file_hash[:16]}...")

    async with session_factory() as session:
        result = await session.execute(

            logger.info(f"  文档已存在, 跳过")
            return 0

    chunks = text_chunker.chunk_document(
        doc.pages,
        metadata={"source_file": Path(file_path).name},
    )
    logger.info(f"  切片完成: {len(chunks)} 个切片")

    # 3. 批量向量化并存储
    inserted = 0
    async with session_factory() as session:
        for i, chunk in enumerate(chunks):
            try:
                # 生成 embedding
                embedding = await embedding_service.embed(chunk.content)
                embedding_str = "[" + ",".join(str(v) for v in embedding) + "]"

                # 计算内容 hash
                import hashlib

                content_hash = hashlib.sha256(chunk.content.encode("utf-8")).hexdigest()

                # 插入数据库
                await session.execute(
                    text("""
                        INSERT INTO document_chunks
                            (document_id, course_id, chunk_index, content, content_hash,
                             embedding, metadata, source_file, page_number, section_title)
                        VALUES
                            (:document_id, :course_id, :chunk_index, :content, :content_hash,
                             :embedding::vector, :metadata, :source_file, :page_number, :section_title)
                    """),
                    {
                        "document_id": doc.file_hash,
                        "course_id": course_id,
                        "chunk_index": chunk.chunk_index,
                        "content": chunk.content,
                        "content_hash": content_hash,
                        "embedding": embedding_str,
                        "metadata": '{"token_count": ' + str(chunk.token_count) + "}",
                        "source_file": chunk.metadata.get("source_file", ""),
                        "page_number": chunk.metadata.get("page_number", 0),
                        "section_title": chunk.section_title,
                    },
                )
                inserted += 1

                if (i + 1) % 10 == 0:
                    logger.info(f"  已处理 {i+1}/{len(chunks)} 个切片")

            except Exception as e:
                logger.error(f"  切片 {i} 处理失败: {e}")
                continue

        await session.commit()

    logger.info(f"  文档处理完成: 插入 {inserted} 个切片")
    return inserted

            total_chunks += count
        except Exception as e:

    logger.info(f"知识库初始化完成! 总共处理 {total_chunks} 个切片")
```

```python
# backend/app/services/source_validator.py
"""
来源验证服务
验证 LLM 生成内容是否基于检索到的来源, 防止幻觉
"""

import re
from dataclasses import dataclass

from app.services.spark_client import spark_client, SparkMessage
from loguru import logger


@dataclass
class ValidationResult:
    is_valid: bool
    confidence: float
    unsupported_claims: list[str]
    supported_claims: list[str]
    correction: str


class SourceValidator:
    """来源验证器"""

    async def validate(
        self, generated_text: str, source_chunks: list[dict]
    ) -> ValidationResult:
        """验证生成内容是否基于来源"""
        if not source_chunks:
            return ValidationResult(
                is_valid=False,
                confidence=0.0,
                unsupported_claims=["无来源可验证"],
                supported_claims=[],
                correction="",
            )

        # 构建来源文本
        sources_text = "\n\n".join(
            f"[来源{i+1}] {s.get('content', '')[:500]}"
            for i, s in enumerate(source_chunks)
        )

        # LLM 验证
        verify_prompt = f"""请验证以下生成内容是否基于提供的参考资料。

生成内容:
{generated_text}

参考资料:
{sources_text}

请分析:
1. 列出内容中有来源支持的陈述 (用 bullet list)
2. 列出内容中无法从来源验证的陈述 (用 bullet list)
3. 整体判断: PASS (大部分有来源支持) 或 FAIL (存在较多无来源内容)
4. 如有不准确之处，给出修正建议

请用以下 JSON 格式返回:
{{"status": "PASS"或"FAIL", "supported": ["..."], "unsupported": ["..."], "correction": "..."}}"""

        messages = [
            SparkMessage(role="system", content="你是一个内容审核助手，严格判断内容是否有来源支持。只返回JSON。"),
            SparkMessage(role="user", content=verify_prompt),
        ]

        try:
            response = await spark_client.chat(messages, temperature=0.1, max_tokens=1000)

            import json

            # 提取 JSON
            json_match = re.search(r"\{.*\}", response.content, re.DOTALL)
            if json_match:
                result = json.loads(json_match.group())
                status = result.get("status", "FAIL")
                supported = result.get("supported", [])
                unsupported = result.get("unsupported", [])
                correction = result.get("correction", "")

                return ValidationResult(
                    is_valid=(status == "PASS"),
                    confidence=0.8 if status == "PASS" else 0.3,
                    unsupported_claims=unsupported,
                    supported_claims=supported,
                    correction=correction,
                )

        except Exception as e:
            logger.error(f"来源验证失败: {e}")

        # 回退: 关键词匹配验证
        return self._keyword_validate(generated_text, source_chunks)

    def _keyword_validate(
        self, generated_text: str, source_chunks: list[dict]
    ) -> ValidationResult:
        """关键词匹配回退验证"""
        # 提取生成文本中的关键实体
        all_source_text = " ".join(s.get("content", "") for s in source_chunks)

        # 简单的关键词覆盖率
        words_generated = set(re.findall(r"[\u4e00-\u9fa5]{2,}|[A-Za-z]{3,}", generated_text))
        words_source = set(re.findall(r"[\u4e00-\u9fa5]{2,}|[A-Za-z]{3,}", all_source_text))

        if not words_generated:
            return ValidationResult(
                is_valid=True, confidence=0.5,
                unsupported_claims=[], supported_claims=[], correction=""
            )

        overlap = words_generated & words_source
        coverage = len(overlap) / len(words_generated) if words_generated else 0

        return ValidationResult(
            is_valid=coverage > 0.4,
            confidence=coverage,
            unsupported_claims=[],
            supported_claims=[],
            correction="",
        )
```

```python
# backend/app/agents/state.py
"""
LangGraph 多智能体状态定义
所有 Agent 共享的状态结构
"""

from typing import Annotated, TypedDict, Optional, Any
from enum import Enum
from dataclasses import dataclass, field

from langgraph.graph import add_messages


class IntentType(str, Enum):
    """用户意图类型"""
    PROFILE_BUILD = "profile_build"       # 构建学习画像
    RESOURCE_GENERATE = "resource_generate"  # 生成学习资源
    LEARNING_PATH = "learning_path"       # 生成学习路径
    ASK_QUESTION = "ask_question"         # 提问/辅导
    CODE_EXAMPLE = "code_example"         # 代码示例
    MIND_MAP = "mind_map"                 # 思维导图
    EXERCISE = "exercise"                 # 练习题
    AUDIO_CONTENT = "audio_content"       # 音频内容
    GENERAL_CHAT = "general_chat"         # 一般对话


class TaskPriority(str, Enum):
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"


class AgentState(TypedDict):
    """多智能体共享状态"""

    # --- 请求信息 ---
    student_id: str
    course_id: str
    session_id: str
    user_message: str

    # --- 意图识别结果 ---
    intent: str                        # IntentType 枚举值
    intent_confidence: float
    intent_params: dict[str, Any]      # 意图相关参数

    # --- 学生画像 ---
    student_profile: dict[str, Any]    # 六维画像 JSONB
    profile_updated: bool              # 画像是否已更新

    # --- 任务规划 ---
    task_plan: list[dict[str, Any]]    # [{agent, params, priority, status}]
    current_task_index: int

    # --- Agent 间消息 ---
    messages: Annotated[list[dict], add_messages]

    # --- 各 Agent 输出 ---
    profile_result: Optional[dict]
    document_result: Optional[dict]    # RAG 检索结果 + 生成内容
    mindmap_result: Optional[dict]     # Mermaid 代码
    exercise_result: Optional[dict]    # 练习题列表
    code_result: Optional[dict]        # 代码示例
    path_result: Optional[dict]        # 学习路径
    video_result: Optional[dict]       # 音频脚本 + URL
    tutor_result: Optional[dict]       # 问答结果

    # --- 聚合结果 ---
    final_response: str                # 最终回复文本
    resources: list[dict]              # 生成的资源列表
    citations: list[str]               # 引用来源

    # --- 任务元数据 ---
    task_id: str
    status: str                        # pending/planning/executing/aggregating/completed/failed
    progress: float                    # 0~1
    current_step: str
    error: Optional[str]

    # --- 流式输出标记 ---
    stream_chunks: list[str]           # SSE 流式数据块
```

```python
# backend/app/agents/master_agent.py
"""
Master Agent - 多智能体编排核心
基于 LangGraph StateGraph 实现:
  意图识别 -> 任务规划 -> 条件路由 -> 结果聚合 -> 响应生成
"""

from typing import Literal
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
from loguru import logger

from app.agents.state import AgentState, IntentType
from app.agents.profile_agent import profile_agent
from app.agents.document_agent import document_agent
from app.agents.mindmap_agent import mindmap_agent
from app.agents.exercise_agent import exercise_agent
from app.agents.code_agent import code_agent
from app.agents.path_agent import path_agent
from app.agents.tutor_agent import tutor_agent
from app.agents.video_agent import video_agent
from app.agents.communicator import MessageBus
from app.services.spark_client import spark_client, SparkMessage


class MasterAgent:
    """Master Agent - 多智能体协调器"""

    def __init__(self):
        self.message_bus = MessageBus()
        self.graph = self._build_graph()

    def _build_graph(self) -> StateGraph:
        """构建 LangGraph 状态图"""
        graph = StateGraph(AgentState)

        # 注册节点
        graph.add_node("intent_recognition", self._intent_recognition)
        graph.add_node("task_planning", self._task_planning)
        graph.add_node("profile_agent", self._run_profile_agent)
        graph.add_node("document_agent", self._run_document_agent)
        graph.add_node("mindmap_agent", self._run_mindmap_agent)
        graph.add_node("exercise_agent", self._run_exercise_agent)
        graph.add_node("code_agent", self._run_code_agent)
        graph.add_node("path_agent", self._run_path_agent)
        graph.add_node("tutor_agent", self._run_tutor_agent)
        graph.add_node("video_agent", self._run_video_agent)
        graph.add_node("result_aggregation", self._result_aggregation)
        graph.add_node("response_generation", self._response_generation)

        # 入口
        graph.set_entry_point("intent_recognition")

        # 意图识别 -> 任务规划
        graph.add_edge("intent_recognition", "task_planning")

        # 任务规划 -> 条件路由
        graph.add_conditional_edges(
            "task_planning",
            self._route_to_agents,
            {
                "profile_agent": "profile_agent",
                "document_agent": "document_agent",
                "mindmap_agent": "mindmap_agent",
                "exercise_agent": "exercise_agent",
                "code_agent": "code_agent",
                "path_agent": "path_agent",
                "tutor_agent": "tutor_agent",
                "video_agent": "video_agent",
                "aggregate": "result_aggregation",
            },
        )

        # 各 Agent 完成后 -> 结果聚合
        for agent_name in [
            "profile_agent", "document_agent", "mindmap_agent",
            "exercise_agent", "code_agent", "path_agent",
            "tutor_agent", "video_agent",
        ]:
            graph.add_conditional_edges(
                agent_name,
                self._check_more_tasks,
                {
                    "next_agent": "task_planning",
                    "aggregate": "result_aggregation",
                },
            )

        # 结果聚合 -> 响应生成 -> END
        graph.add_edge("result_aggregation", "response_generation")
        graph.add_edge("response_generation", END)

        return graph.compile(checkpointer=MemorySaver())

    # ------------------------------------------------------------------
    # 节点实现
    # ------------------------------------------------------------------

    async def _intent_recognition(self, state: AgentState) -> dict:
        """意图识别节点"""
        logger.info(f"[Master] 意图识别: {state['user_message'][:50]}...")

        system_prompt = """你是一个学习平台的意图识别助手。
根据用户消息，判断其意图类型，返回 JSON 格式:

意图类型:
- profile_build: 构建/更新学习画像 (如填写问卷、了解学习风格)
- resource_generate: 生成学习资源 (如学习材料、知识点讲解)
- learning_path: 生成学习路径 (如学习计划、课程路线)
- ask_question: 提问/辅导 (如知识点疑问、概念解释)
- code_example: 请求代码示例 (如代码演示、编程示例)
- mind_map: 思维导图 (如知识结构、概念图)
- exercise: 练习题 (如习题、测试题)
- audio_content: 音频内容 (如语音讲解、音频学习)
- general_chat: 一般对话/闲聊

返回格式: {"intent": "...", "confidence": 0.0~1.0, "params": {...}}
params 中提取关键参数: topic(主题), difficulty(难度), count(数量)等"""

        messages = [
            SparkMessage(role="system", content=system_prompt),
            SparkMessage(role="user", content=state["user_message"]),
        ]

        try:
            response = await spark_client.chat(messages, temperature=0.1, max_tokens=500)

            import json
            import re

            json_match = re.search(r"\{.*\}", response.content, re.DOTALL)
            if json_match:
                result = json.loads(json_match.group())
                intent = result.get("intent", "general_chat")
                confidence = result.get("confidence", 0.5)
                params = result.get("params", {})
            else:
                intent = "general_chat"
                confidence = 0.3
                params = {}

        except Exception as e:
            logger.error(f"意图识别失败: {e}")
            intent = "general_chat"
            confidence = 0.3
            params = {}

        logger.info(f"[Master] 意图: {intent} (置信度: {confidence})")

        return {
            "intent": intent,
            "intent_confidence": confidence,
            "intent_params": params,
            "status": "planning",
            "current_step": "意图识别完成",
            "progress": 0.15,
        }

    async def _task_planning(self, state: AgentState) -> dict:
        """任务规划节点: 根据意图生成任务计划"""
        logger.info(f"[Master] 任务规划: intent={state['intent']}")

        intent = state["intent"]
        params = state.get("intent_params", {})

        # 任务计划模板
        task_templates = {
            "profile_build": [
                {"agent": "profile_agent", "action": "build_profile", "priority": "high"}
            ],
            "resource_generate": [
                {"agent": "document_agent", "action": "generate_explanation", "priority": "high"},
                {"agent": "mindmap_agent", "action": "generate_mindmap", "priority": "medium"},
                {"agent": "exercise_agent", "action": "generate_exercises", "priority": "medium"},
            ],
            "learning_path": [
                {"agent": "path_agent", "action": "generate_path", "priority": "high"}
            ],
            "ask_question": [
                {"agent": "tutor_agent", "action": "answer_question", "priority": "high"}
            ],
            "code_example": [
                {"agent": "code_agent", "action": "generate_code", "priority": "high"}
            ],
            "mind_map": [
                {"agent": "mindmap_agent", "action": "generate_mindmap", "priority": "high"}
            ],
            "exercise": [
                {"agent": "exercise_agent", "action": "generate_exercises", "priority": "high"}
            ],
            "audio_content": [
                {"agent": "document_agent", "action": "generate_script", "priority": "high"},
                {"agent": "video_agent", "action": "generate_audio", "priority": "high"},
            ],
            "general_chat": [
                {"agent": "tutor_agent", "action": "general_response", "priority": "low"}
            ],
        }

        tasks = task_templates.get(intent, task_templates["general_chat"])

        # 注入参数
        for task in tasks:
            task["params"] = params
            task["status"] = "pending"

        return {
            "task_plan": tasks,
            "current_task_index": 0,
            "status": "executing",
            "current_step": f"规划完成, {len(tasks)} 个任务待执行",
            "progress": 0.25,
        }

    def _route_to_agents(self, state: AgentState) -> str:
        """条件路由: 根据当前任务路由到对应 Agent"""
        tasks = state.get("task_plan", [])
        idx = state.get("current_task_index", 0)

        if idx >= len(tasks):
            return "aggregate"

        current_task = tasks[idx]
        agent_name = current_task.get("agent", "")

        route_map = {
            "profile_agent": "profile_agent",
            "document_agent": "document_agent",
            "mindmap_agent": "mindmap_agent",
            "exercise_agent": "exercise_agent",
            "code_agent": "code_agent",
            "path_agent": "path_agent",
            "tutor_agent": "tutor_agent",
            "video_agent": "video_agent",
        }

        route = route_map.get(agent_name)
        if route is None:
            logger.warning(f"[Master] 未知 Agent: {agent_name}, 跳过")
            return "aggregate"

        logger.info(f"[Master] 路由到: {agent_name}")
        return route

    def _check_more_tasks(self, state: AgentState) -> str:
        """检查是否还有待执行的任务"""
        tasks = state.get("task_plan", [])
        idx = state.get("current_task_index", 0)

        if idx + 1 < len(tasks):
            return "next_agent"
        return "aggregate"

    # ------------------------------------------------------------------
    # Agent 执行节点
    # ------------------------------------------------------------------

    async def _run_profile_agent(self, state: AgentState) -> dict:

        logger.info("[Master] 执行 Profile Agent")

        # 更新任务进度
        tasks = list(state.get("task_plan", []))
        idx = state.get("current_task_index", 0)
        if idx < len(tasks):
            tasks[idx] = {**tasks[idx], "status": "completed"}

        return {
            "profile_result": result,
            "task_plan": tasks,
            "current_task_index": idx + 1,
            "progress": state.get("progress", 0) + 0.1,
            "current_step": "画像分析完成",
            "messages": [{"role": "system", "content": f"Profile Agent 完成: {result.get('summary', '')}"}],
        }

    async def _result_aggregation(self, state: AgentState) -> dict:
        """结果聚合节点: 汇总所有 Agent 输出"""

        resources = []
        citations = []

        # 聚合画像结果

            pr = state["profile_result"]
            response_parts.append(f"**学习画像分析完成**\n{pr.get('summary', '')}")

        # 聚合文档结果
        if state.get("document_result"):
            dr = state["document_result"]
            response_parts.append(dr.get("content", ""))
            if dr.get("citations"):
                citations.extend(dr["citations"])
            resources.append({
                "type": "explanation",
                "title": dr.get("title", "知识点讲解"),
                "content": dr.get("content", ""),
            })

        # 聚合思维导图
        if state.get("mindmap_result"):
            mr = state["mindmap_result"]
            response_parts.append(f"**思维导图**\n```mermaid\n{mr.get('mermaid_code', '')}\n```")
            resources.append({
                "type": "mindmap",
                "title": mr.get("title", "知识结构图"),
                "mermaid_code": mr.get("mermaid_code", ""),
            })

        # 聚合练习题
        if state.get("exercise_result"):
            er = state["exercise_result"]
            exercises = er.get("exercises", [])
            response_parts.append(f"**生成 {len(exercises)} 道练习题**")
            resources.append({
                "type": "exercise",
                "title": er.get("title", "练习题"),
                "exercises": exercises,
            })

        # 聚合代码示例
        if state.get("code_result"):
            cr = state["code_result"]
            response_parts.append(f"**代码示例**\n```{cr.get('language', 'python')}\n{cr.get('code', '')}\n```")
            resources.append({
                "type": "code",
                "title": cr.get("title", "代码示例"),
                "code": cr.get("code", ""),
                "language": cr.get("language", "python"),
            })

        # 聚合学习路径
        if state.get("path_result"):
            pr = state["path_result"]
            response_parts.append("**学习路径已生成**")
            resources.append({
                "type": "path",
                "title": pr.get("title", "学习路径"),
                "nodes": pr.get("nodes", []),
                "edges": pr.get("edges", []),
            })

        # 聚合问答结果
        if state.get("tutor_result"):
            tr = state["tutor_result"]
            response_parts.append(tr.get("answer", ""))
            if tr.get("citations"):
                citations.extend(tr["citations"])

        # 聚合音频结果
        if state.get("video_result"):
            vr = state["video_result"]
            response_parts.append(f"**音频内容已生成** 🔊")
            resources.append({
                "type": "audio",
                "title": vr.get("title", "音频讲解"),
                "audio_url": vr.get("audio_url", ""),
                "script": vr.get("script", ""),
            })

        final_response = "\n\n---\n\n".join(response_parts) if response_parts else "任务处理完成。"

        return {
            "final_response": final_response,
            "resources": resources,
            "citations": citations,
            "status": "aggregating",
            "progress": 0.9,
            "current_step": "结果聚合完成",
        }

    async def _response_generation(self, state: AgentState) -> dict:
        """响应生成节点: 最终润色"""
        logger.info("[Master] 响应生成")

        return {
            "status": "completed",
            "progress": 1.0,
            "current_step": "完成",
        }

    # ------------------------------------------------------------------
    # 公开入口
    # ------------------------------------------------------------------

    async def run(self, initial_state: dict) -> dict:
        """执行多智能体流程"""
        import uuid

        task_id = initial_state.get("task_id", str(uuid.uuid4()))

        # 确保必要字段存在
        full_state = {
            "student_id": "",
            "course_id": "",
            "session_id": "",
            "user_message": "",
            "intent": "",
            "intent_confidence": 0.0,
            "intent_params": {},
            "student_profile": {},
            "profile_updated": False,
            "task_plan": [],
            "current_task_index": 0,
            "messages": [],
            "profile_result": None,
            "document_result": None,
            "mindmap_result": None,
            "exercise_result": None,
            "code_result": None,
            "path_result": None,
            "video_result": None,
            "tutor_result": None,
            "final_response": "",
            "resources": [],
            "citations": [],
            "task_id": task_id,
            "status": "pending",
            "progress": 0.0,
            "current_step": "",
            "error": None,
            "stream_chunks": [],
            **initial_state,
        }

        try:
            config = {"configurable": {"thread_id": task_id}}
            result = await self.graph.ainvoke(full_state, config=config)
            return result

        except Exception as e:
            logger.error(f"[Master] 执行失败: {e}")
            return {**full_state, "status": "failed", "error": str(e)}
```

```python
class ProfileAgent:
    """画像 Agent"""

    async def execute(self, state: dict) -> dict:
        """执行画像相关任务"""
        action = state.get("intent_params", {}).get("action", "build")

        if action == "update":
            return await self._update_profile(state)

            return await self._generate_follow_up(state)

    async def _build_profile(self, state: dict) -> dict:
```

```python
"""
Path Agent - 自适应学习路径生成
核心算法: 拓扑排序 + 难度优先 + 弱项优先
"""

from collections import defaultdict, deque
from typing import Any

from loguru import logger

from app.services.spark_client import spark_client, SparkMessage


class PathAgent:
    """学习路径 Agent"""

    async def execute(self, state: dict) -> dict:
        """生成学习路径"""
        course_id = state.get("course_id", "")
        profile = state.get("student_profile", {})

        # 1. 获取知识点图谱
        knowledge_graph = await self._load_knowledge_graph(course_id)

        # 2. 拓扑排序
        sorted_kps = self._topological_sort(knowledge_graph)

        # 3. 根据画像调整优先级
        prioritized_kps = self._prioritize_by_profile(sorted_kps, profile)

        # 4. 生成每日学习计划
        daily_plan = self._generate_daily_plan(prioritized_kps, profile)

        # 5. 构建路径数据
        nodes, edges = self._build_path_data(prioritized_kps, knowledge_graph)

        # 6. LLM 生成路径描述
        path_desc = await self._generate_path_description(
            prioritized_kps, profile, daily_plan
        )

        return {
            "title": "个性化学习路径",
            "nodes": nodes,
            "edges": edges,
            "daily_plan": daily_plan,
            "description": path_desc,
            "total_hours": sum(kp.get("estimated_hours", 2) for kp in prioritized_kps),
            "total_steps": len(prioritized_kps),
        }

    async def _load_knowledge_graph(self, course_id: str) -> dict:
        """从数据库加载知识图谱"""
        from app.db.session import async_session_factory
        from sqlalchemy import text

        async with async_session_factory() as db:
            result = await db.execute(
                text("""
                    SELECT id, name, description, difficulty, category,
                           prerequisites, keywords, order_index
                    FROM knowledge_points
                    WHERE course_id = :course_id
                    ORDER BY order_index
                """),
                {"course_id": course_id},
            )

            kps = []
            for row in result.mappings():
                kps.append({
                    "id": str(row["id"]),
                    "name": row["name"],
                    "description": row["description"] or "",
                    "difficulty": row["difficulty"] or "intermediate",
                    "category": row["category"] or "",
                    "prerequisites": [str(p) for p in (row["prerequisites"] or [])],
                    "keywords": row["keywords"] or [],
                    "order_index": row["order_index"] or 0,
                })

        # 构建邻接表
        graph = {"nodes": {}, "edges": []}
        for kp in kps:
            graph["nodes"][kp["id"]] = kp
            for prereq_id in kp["prerequisites"]:
                graph["edges"].append({
                    "from": prereq_id,
                    "to": kp["id"],
                    "relation": "prerequisite",
                })

        return graph

    def _topological_sort(self, graph: dict) -> list[dict]:
        """拓扑排序: 确保前置知识点先学"""
        nodes = graph["nodes"]
        edges = graph["edges"]

        # 计算入度
        in_degree = defaultdict(int)
        adjacency = defaultdict(list)

        for node_id in nodes:
            in_degree[node_id] = 0

        for edge in edges:
            from_id, to_id = edge["from"], edge["to"]
            if from_id in nodes and to_id in nodes:
                adjacency[from_id].append(to_id)
                in_degree[to_id] += 1

        # Kahn 算法
        queue = deque()
        for node_id in nodes:
            if in_degree[node_id] == 0:
                queue.append(node_id)

        sorted_order = []
        while queue:
            current = queue.popleft()
            sorted_order.append(nodes[current])

            for neighbor in adjacency[current]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)

        # 处理环 (如果存在)
        if len(sorted_order) < len(nodes):
            remaining = [nodes[nid] for nid in nodes if nid not in {n["id"] for n in sorted_order}]
            sorted_order.extend(remaining)

        return sorted_order

    def _prioritize_by_profile(
        self, sorted_kps: list[dict], profile: dict
    ) -> list[dict]:
        """根据画像调整优先级: 弱项优先 + 已掌握跳过"""
        dims = profile.get("dimensions", {})
        mastery = dims.get("knowledge_mastery", {})
        weak_topics = dims.get("weak_topics", [])

        for kp in sorted_kps:
            kp_name = kp["name"]
            mastery_score = mastery.get(kp_name, 0)

            # 已掌握 (>= 0.8) 降低优先级
            if mastery_score >= 0.8:
                kp["priority"] = "low"
                kp["estimated_hours"] = 0.5  # 快速回顾
            # 薄弱环节提高优先级
            elif kp_name in weak_topics:
                kp["priority"] = "high"
                kp["estimated_hours"] = self._estimate_hours(kp["difficulty"], boost=True)
            else:
                kp["priority"] = "medium"
                kp["estimated_hours"] = self._estimate_hours(kp["difficulty"])

        # 排序: 高优先级在前, 同优先级按拓扑序
        priority_order = {"high": 0, "medium": 1, "low": 2}
        sorted_kps.sort(key=lambda kp: priority_order.get(kp.get("priority", "medium"), 1))

        return sorted_kps

    def _estimate_hours(self, difficulty: str, boost: bool = False) -> float:
        """估算学习时间"""
        base = {"beginner": 1.5, "intermediate": 2.5, "advanced": 4.0}
        hours = base.get(difficulty, 2.0)
        return hours * 1.3 if boost else hours

    def _generate_daily_plan(
        self, prioritized_kps: list[dict], profile: dict
    ) -> list[dict]:
        """生成每日学习计划"""
        pace = profile.get("dimensions", {}).get("learning_pace", {})
        daily_hours = pace.get("daily_hours", 2.0)

        plan = []
        current_day = 1
        current_day_hours = 0
        current_day_items = []

        for kp in prioritized_kps:
            kp_hours = kp.get("estimated_hours", 2.0)

            if current_day_hours + kp_hours > daily_hours and current_day_items:
                plan.append({
                    "day": current_day,
                    "items": current_day_items,
                    "total_hours": round(current_day_hours, 1),
                })
                current_day += 1
                current_day_items = []
                current_day_hours = 0

            current_day_items.append({
                "kp_id": kp["id"],
                "kp_name": kp["name"],
                "difficulty": kp["difficulty"],
                "estimated_hours": kp_hours,
                "priority": kp.get("priority", "medium"),
            })
            current_day_hours += kp_hours

        # 最后一天
        if current_day_items:
            plan.append({
                "day": current_day,
                "items": current_day_items,
                "total_hours": round(current_day_hours, 1),
            })

        return plan

    def _build_path_data(
        self, prioritized_kps: list[dict], graph: dict
    ) -> tuple[list, list]:
        """构建路径可视化数据"""
        nodes = []
        for i, kp in enumerate(prioritized_kps):
            nodes.append({
                "id": kp["id"],
                "kp_name": kp["name"],
                "order": i,
                "difficulty": kp["difficulty"],
                "status": "pending",
                "estimated_hours": kp.get("estimated_hours", 2.0),
                "priority": kp.get("priority", "medium"),
                "category": kp.get("category", ""),
            })

        edges = []
        for edge in graph["edges"]:
            # 只保留路径中节点之间的边
            node_ids = {n["id"] for n in nodes}
            if edge["from"] in node_ids and edge["to"] in node_ids:
                edges.append(edge)

        return nodes, edges

    async def _generate_path_description(
        self, prioritized_kps: list[dict], profile: dict, daily_plan: list[dict]
    ) -> str:
        """LLM 生成路径描述"""
        weak_topics = profile.get("dimensions", {}).get("weak_topics", [])
        total_days = len(daily_plan)

        kp_list = "\n".join(
            f"- {kp['name']} ({kp['difficulty']}, 优先级: {kp.get('priority', 'medium')})"
            for kp in prioritized_kps[:10]
        )

        messages = [
            SparkMessage(
                role="system",
                content="你是一个学习规划师。请用简洁友好的语言描述学习路径计划。",
            ),
            SparkMessage(
                role="user",
                content=(
                    f"知识点列表:\n{kp_list}\n"
                    f"薄弱环节: {', '.join(weak_topics[:3])}\n"
                    f"预计 {total_days} 天完成\n\n"
                    "请生成一段简洁的学习路径描述 (100-200字)。"
                ),
            ),
        ]

        response = await spark_client.chat(messages, temperature=0.5, max_tokens=300)
        return response.content
```

```python
"""
Agent 通信机制
- MessageBus: 基于 asyncio.Queue 的 pub/sub 消息总线
- AgentCommunicator: Agent 间请求/响应通信
"""

import asyncio
from typing import Any, Callable, Awaitable
from dataclasses import dataclass, field
from enum import Enum

from loguru import logger


class MessageType(str, Enum):
    REQUEST = "request"
    RESPONSE = "response"
    EVENT = "event"
    BROADCAST = "broadcast"


@dataclass
class AgentMessage:
    msg_type: MessageType
    sender: str
    receiver: str
    action: str
    payload: dict
    correlation_id: str = ""
    reply_to: str = ""


class MessageBus:
    """Agent 间消息总线 (Pub/Sub 模式)"""

    def __init__(self):
        self._subscribers: dict[str, list[Callable]] = {}
        self._queues: dict[str, asyncio.Queue] = {}
        self._history: list[AgentMessage] = []

    def subscribe(self, topic: str, handler: Callable[[AgentMessage], Awaitable[None]]):
        """订阅主题"""
        if topic not in self._subscribers:
            self._subscribers[topic] = []
        self._subscribers[topic].append(handler)
        logger.debug(f"[MessageBus] 订阅: {topic}")

    def unsubscribe(self, topic: str, handler: Callable):
        """取消订阅"""
        if topic in self._subscribers:
            self._subscribers[topic] = [
                h for h in self._subscribers[topic] if h != handler
            ]

    async def publish(self, topic: str, message: AgentMessage):
        """发布消息"""
        self._history.append(message)
        handlers = self._subscribers.get(topic, [])

        if not handlers:
            logger.debug(f"[MessageBus] 主题 {topic} 无订阅者")
            return

        tasks = [handler(message) for handler in handlers]
        await asyncio.gather(*tasks, return_exceptions=True)

    async def request(
        self, sender: str, receiver: str, action: str, payload: dict
    ) -> dict:
        """请求/响应模式"""
        correlation_id = f"{sender}_{receiver}_{action}_{id(payload)}"
        response_queue = asyncio.Queue()
        self._queues[correlation_id] = response_queue

        message = AgentMessage(
            msg_type=MessageType.REQUEST,
            sender=sender,
            receiver=receiver,
            action=action,
            payload=payload,
            correlation_id=correlation_id,
        )

        await self.publish(f"agent.{receiver}", message)

        try:
            response = await asyncio.wait_for(response_queue.get(), timeout=30.0)
            return response
        except asyncio.TimeoutError:
            logger.warning(f"[MessageBus] 请求超时: {correlation_id}")
            return {"error": "timeout"}
        finally:
            self._queues.pop(correlation_id, None)

    async def respond(self, correlation_id: str, payload: dict):
        """发送响应"""
        queue = self._queues.get(correlation_id)
        if queue:
            await queue.put(payload)

    def get_history(self, limit: int = 50) -> list[AgentMessage]:
        """获取消息历史"""
        return self._history[-limit:]


class AgentCommunicator:
    """Agent 通信器 - 封装常用通信模式"""

    def __init__(self, agent_name: str, bus: MessageBus):
        self.agent_name = agent_name
        self.bus = bus

    async def request_profile(self, student_id: str) -> dict:
        """请求学生画像"""
        return await self.bus.request(
            sender=self.agent_name,
            receiver="profile_agent",
            action="get_profile",
            payload={"student_id": student_id},
        )

    async def request_rag(self, query: str, course_id: str) -> dict:
        """请求 RAG 检索"""
        return await self.bus.request(
            sender=self.agent_name,
            receiver="document_agent",
            action="retrieve",
            payload={"query": query, "course_id": course_id},
        )

    async def broadcast_event(self, event_type: str, data: dict):
        """广播事件"""
        message = AgentMessage(
            msg_type=MessageType.BROADCAST,
            sender=self.agent_name,
            receiver="*",
            action=event_type,
            payload=data,
        )
        await self.bus.publish("events", message)

    async def notify_progress(self, task_id: str, progress: float, step: str):
        """通知任务进度"""
        message = AgentMessage(
            msg_type=MessageType.EVENT,
            sender=self.agent_name,
            receiver="master_agent",
            action="progress_update",
            payload={
                "task_id": task_id,
                "progress": progress,
                "step": step,
            },
        )
        await self.bus.publish("progress", message)
```

```python
"""
画像更新触发器
各种学习行为触发画像的自动更新
"""

from enum import Enum
from typing import Optional
from loguru import logger

from app.services.cache_service import cache_service


class TriggerType(str, Enum):
    ON_LEARNING_ACTION = "on_learning_action"         # 学习行为 (浏览/完成)
    ON_EXERCISE_COMPLETED = "on_exercise_completed"    # 练习完成
    ON_FEEDBACK_PROVIDED = "on_feedback_provided"     # 反馈提交
    ON_CONVERSATION = "on_conversation"               # 对话交互
    ON_RESOURCE_INTERACTION = "on_resource_interaction"  # 资源交互


class ProfileTrigger:
    """画像更新触发器"""

    # 触发条件阈值
    EXERCISE_TRIGGER_COUNT = 3       # 每完成 3 道题触发
    INTERACTION_TRIGGER_COUNT = 5    # 每 5 次交互触发
    FEEDBACK_TRIGGER_COUNT = 1       # 每次反馈立即触发

    def __init__(self):
        self._counters: dict[str, dict[str, int]] = {}  # student_id -> {trigger: count}

    def _get_counter(self, student_id: str, trigger: str) -> int:
        return self._counters.get(student_id, {}).get(trigger, 0)

    def _increment_counter(self, student_id: str, trigger: str) -> int:
        if student_id not in self._counters:
            self._counters[student_id] = {}
        self._counters[student_id][trigger] = (
            self._counters[student_id].get(trigger, 0) + 1
        )
        return self._counters[student_id][trigger]

    def should_update(self, student_id: str, trigger_type: TriggerType) -> bool:
        """判断是否应该触发画像更新"""
        count = self._increment_counter(student_id, trigger_type.value)

        threshold_map = {
            TriggerType.ON_EXERCISE_COMPLETED: self.EXERCISE_TRIGGER_COUNT,
            TriggerType.ON_RESOURCE_INTERACTION: self.INTERACTION_TRIGGER_COUNT,
            TriggerType.ON_FEEDBACK_PROVIDED: self.FEEDBACK_TRIGGER_COUNT,
            TriggerType.ON_CONVERSATION: self.INTERACTION_TRIGGER_COUNT,
            TriggerType.ON_LEARNING_ACTION: self.INTERACTION_TRIGGER_COUNT,
        }

        threshold = threshold_map.get(trigger_type, 5)
        return count >= threshold

    def reset_counter(self, student_id: str, trigger_type: TriggerType):
        """重置计数器"""
        if student_id in self._counters:
            self._counters[student_id][trigger_type.value] = 0

    async def on_learning_action(
        self,
        student_id: str,
        course_id: str,
        action_data: dict,
    ) -> Optional[dict]:
        """学习行为触发"""
        trigger_type = TriggerType.ON_LEARNING_ACTION

        # 记录学习行为
        await self._record_action(student_id, course_id, action_data)

        if self.should_update(student_id, trigger_type):
            self.reset_counter(student_id, trigger_type)
            return await self._trigger_update(student_id, course_id, action_data)

        return None

    async def on_exercise_completed(
        self,
        student_id: str,
        course_id: str,
        exercise_data: dict,
    ) -> Optional[dict]:
        """练习完成触发"""
        trigger_type = TriggerType.ON_EXERCISE_COMPLETED

        # 记录练习结果
        await self._record_exercise(student_id, course_id, exercise_data)

        if self.should_update(student_id, trigger_type):
            self.reset_counter(student_id, trigger_type)
            return await self._trigger_update(
                student_id, course_id, exercise_data, trigger_type
            )

        return None

    async def on_feedback_provided(
        self,
        student_id: str,
        course_id: str,
        feedback_data: dict,
    ) -> Optional[dict]:
        """反馈触发 (立即更新)"""
        return await self._trigger_update(
            student_id, course_id, feedback_data, TriggerType.ON_FEEDBACK_PROVIDED
        )

    async def on_conversation(
        self,
        student_id: str,
        course_id: str,
        conversation_data: dict,
    ) -> Optional[dict]:
        """对话触发"""
        trigger_type = TriggerType.ON_CONVERSATION

        if self.should_update(student_id, trigger_type):
            self.reset_counter(student_id, trigger_type)
            return await self._trigger_update(
                student_id, course_id, conversation_data, trigger_type
            )

        return None

    async def on_resource_interaction(
        self,
        student_id: str,
        course_id: str,
        interaction_data: dict,
    ) -> Optional[dict]:
        """资源交互触发"""
        trigger_type = TriggerType.ON_RESOURCE_INTERACTION

        if self.should_update(student_id, trigger_type):
            self.reset_counter(student_id, trigger_type)
            return await self._trigger_update(
                student_id, course_id, interaction_data, trigger_type
            )

        return None

    async def _record_action(self, student_id: str, course_id: str, data: dict):
        """记录学习行为到数据库"""
        from app.db.session import async_session_factory
        from sqlalchemy import text

            await db.execute(
                text("""
                    INSERT INTO learning_records
                        (student_id, course_id, kp_id, resource_id, action, duration, detail)
                    VALUES
                        (:student_id, :course_id, :kp_id, :resource_id, :action, :duration, :detail)
                """),
                {

    async def _record_exercise(self, student_id: str, course_id: str, data: dict):
        """记录练习结果"""
        from app.db.session import async_session_factory

                    SELECT action, kp_id, score, detail, created_at

                    LIMIT 20

            )

        await cache_service.invalidate_profile(student_id, course_id)
```

```python
"""
学习路径生成算法
核心: 拓扑排序 + 难度优先 + 弱项优先 + 每日计划
"""

from collections import defaultdict, deque
from dataclasses import dataclass, field
from typing import Optional

from loguru import logger


@dataclass
class KnowledgeNode:
    id: str
    name: str
    difficulty: str          # beginner / intermediate / advanced
    category: str
    prerequisites: list[str] = field(default_factory=list)
    estimated_hours: float = 2.0
    mastery_score: float = 0.0  # 学生掌握度 0~1
    is_weak: bool = False
    priority_score: float = 0.0
    status: str = "pending"     # pending / in_progress / completed


@dataclass
class PathEdge:
    from_id: str
    to_id: str
    relation: str = "prerequisite"


@dataclass
class DailyPlan:
    day: int
    items: list[KnowledgeNode]
    total_hours: float


class LearningPathGenerator:
    """学习路径生成器"""

    def __init__(
        self,
        daily_hours: float = 2.0,
        max_days: int = 30,
        weak_boost: float = 1.5,
        mastered_threshold: float = 0.8,
    ):
        self.daily_hours = daily_hours
        self.max_days = max_days
        self.weak_boost = weak_boost
        self.mastered_threshold = mastered_threshold

    def generate(
        self,
        nodes: list[KnowledgeNode],
        edges: list[PathEdge],
    ) -> tuple[list[KnowledgeNode], list[DailyPlan]]:
        """
        生成学习路径
        返回: (排序后的知识点列表, 每日计划)
        """
        # 第一步: 拓扑排序 (确保前置知识点先学)
        sorted_nodes = self._topological_sort(nodes, edges)

        # 第二步: 计算优先级分数
        self._calculate_priority(sorted_nodes)

        # 第三步: 按优先级分组排序 (同优先级内保持拓扑序)
        grouped = self._group_by_priority(sorted_nodes)

        # 第四步: 过滤已掌握的知识点 (移到最后快速回顾)
        main_nodes, review_nodes = self._separate_mastered(grouped)

        # 第五步: 生成每日计划
        daily_plans = self._create_daily_plans(main_nodes, review_nodes)

        return main_nodes + review_nodes, daily_plans

    def _topological_sort(
        self, nodes: list[KnowledgeNode], edges: list[PathEdge]
    ) -> list[KnowledgeNode]:
        """Kahn 算法拓扑排序"""
        node_map = {n.id: n for n in nodes}

        # 构建邻接表和入度表
        in_degree: dict[str, int] = {n.id: 0 for n in nodes}
        adjacency: dict[str, list[str]] = defaultdict(list)

        for edge in edges:
            if edge.from_id in node_map and edge.to_id in node_map:
                adjacency[edge.from_id].append(edge.to_id)
                in_degree[edge.to_id] += 1

        # BFS 拓扑排序
        queue: deque[str] = deque()
        for node_id, degree in in_degree.items():
            if degree == 0:
                queue.append(node_id)

        sorted_ids: list[str] = []
        while queue:
            current = queue.popleft()
            sorted_ids.append(current)

            for neighbor in adjacency[current]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)

        # 处理环 (不应该出现，但防御性处理)
        if len(sorted_ids) < len(nodes):
            remaining = [n.id for n in nodes if n.id not in set(sorted_ids)]
            sorted_ids.extend(remaining)
            logger.warning(f"检测到循环依赖: {remaining}")

        return [node_map[nid] for nid in sorted_ids if nid in node_map]

    def _calculate_priority(self, nodes: list[KnowledgeNode]):
        """计算每个知识点的优先级分数"""
        difficulty_weight = {"beginner": 1.0, "intermediate": 2.0, "advanced": 3.0}

        for node in nodes:
            score = 0.0

            # 难度权重 (低难度优先)
            score += (4.0 - difficulty_weight.get(node.difficulty, 2.0)) * 2.0

            # 弱项加权
            if node.is_weak:
                score += self.weak_boost * 3.0

            # 掌握度低优先
            score += (1.0 - node.mastery_score) * 4.0

            # 前置知识点数量 (叶子节点优先)
            if not any(
                edge.to_id == node.id
                for edge in []  # 这里简化，实际需要传入 edges
            ):
                score += 1.0

            node.priority_score = round(score, 2)

    def _group_by_priority(
        self, nodes: list[KnowledgeNode]
    ) -> list[KnowledgeNode]:
        """按优先级分组排序"""
        # 高优先级在前，同优先级保持原序 (拓扑序)
        return sorted(nodes, key=lambda n: -n.priority_score)

    def _separate_mastered(
        self, nodes: list[KnowledgeNode]
    ) -> tuple[list[KnowledgeNode], list[KnowledgeNode]]:
        """分离已掌握和待学习"""
        main = []
        review = []

        for node in nodes:
            if node.mastery_score >= self.mastered_threshold:
                node.estimated_hours = 0.5  # 快速回顾
                review.append(node)
            else:
                main.append(node)

        return main, review

    def _create_daily_plans(
        self,
        main_nodes: list[KnowledgeNode],
        review_nodes: list[KnowledgeNode],
    ) -> list[DailyPlan]:
        """生成每日学习计划"""
        all_nodes = main_nodes + review_nodes
        plans: list[DailyPlan] = []

        current_day = 1
        current_hours = 0.0
        current_items: list[KnowledgeNode] = []

        for node in all_nodes:
            # 如果当天时间不够，开启新一天
            if (
                current_hours + node.estimated_hours > self.daily_hours
                and current_items
            ):
                plans.append(
                    DailyPlan(
                        day=current_day,
                        items=current_items.copy(),
                        total_hours=round(current_hours, 1),
                    )
                )
                current_day += 1
                current_hours = 0.0
                current_items = []

                if current_day > self.max_days:
                    break

            current_items.append(node)
            current_hours += node.estimated_hours

        # 最后一天
        if current_items and current_day <= self.max_days:
            plans.append(
                DailyPlan(
                    day=current_day,
                    items=current_items,
                    total_hours=round(current_hours, 1),
                )
            )

        return plans


def build_nodes_from_profile(
    knowledge_points: list[dict],
    profile: dict,
) -> tuple[list[KnowledgeNode], list[PathEdge]]:
    """从数据库数据和画像构建节点/边"""
    dims = profile.get("dimensions", {})
    mastery = dims.get("knowledge_mastery", {})
    weak_topics = set(dims.get("weak_topics", []))

    nodes = []
    edges = []

    for kp in knowledge_points:
        kp_name = kp["name"]
        mastery_score = mastery.get(kp_name, 0.0)

        node = KnowledgeNode(
            id=kp["id"],
            name=kp_name,
            difficulty=kp.get("difficulty", "intermediate"),
            category=kp.get("category", ""),
            prerequisites=kp.get("prerequisites", []),
            estimated_hours=_estimate_hours(kp.get("difficulty", "intermediate")),
            mastery_score=mastery_score,
            is_weak=kp_name in weak_topics,
        )
        nodes.append(node)

        # 构建边
        for prereq_id in node.prerequisites:
            edges.append(PathEdge(from_id=prereq_id, to_id=node.id))

    return nodes, edges


def _estimate_hours(difficulty: str) -> float:
    """估算学习时间"""
    return {"beginner": 1.5, "intermediate": 2.5, "advanced": 4.0}.get(
        difficulty, 2.0
    )
```

```python
"""
```

```python
支持: 实时对话、任务进度订阅、多客户端

import json

from fastapi import APIRouter, WebSocket, WebSocketDisconnect

router = APIRouter(tags=["websocket"])


class ConnectionManager:
    """WebSocket 连接管理器"""

    def __init__(self):
        self._connections: dict[str, dict[str, WebSocket]] = {}
        # {student_id: {connection_id: WebSocket}}

    async def connect(
        self, websocket: WebSocket, student_id: str, connection_id: str
    ):
        await websocket.accept()
        if student_id not in self._connections:
            self._connections[student_id] = {}
        self._connections[student_id][connection_id] = websocket
        logger.info(f"[WS] 连接建立: student={student_id}, conn={connection_id}")

    def disconnect(self, student_id: str, connection_id: str):
        if student_id in self._connections:
            self._connections[student_id].pop(connection_id, None)
            if not self._connections[student_id]:
                del self._connections[student_id]
        logger.info(f"[WS] 连接断开: student={student_id}, conn={connection_id}")

    async def send_to_student(self, student_id: str, message: dict):
        """发送给指定学生的所有连接"""
        connections = self._connections.get(student_id, {})
        disconnected = []

        for conn_id, ws in connections.items():
            try:
                await ws.send_json(message)
            except Exception:
                disconnected.append(conn_id)

        for conn_id in disconnected:
            self.disconnect(student_id, conn_id)

    async def broadcast(self, message: dict):
        """广播给所有连接"""
        for student_id in list(self._connections.keys()):
            await self.send_to_student(student_id, message)

@router.websocket("/ws/{student_id}")
async def websocket_endpoint(websocket: WebSocket, student_id: str):
    """WebSocket 端点"""
    import uuid

    connection_id = str(uuid.uuid4())[:8]
    await ws_manager.connect(websocket, student_id, connection_id)

    # 启动进度监听任务
    progress_task = None

    try:
        while True:
            data = await websocket.receive_json()

            msg_type = data.get("type", "")

            if msg_type == "subscribe_task":
                # 订阅任务进度
                task_id = data.get("task_id", "")
                if task_id:
                    progress_task = asyncio.create_task(
                        _listen_task_progress(
                            websocket, student_id, task_id
                        )
                    )

            elif msg_type == "chat":
                # 处理聊天消息 (转发到 Agent)
                message = data.get("message", "")
                course_id = data.get("course_id", "")
                session_id = data.get("session_id", "")

                # 异步处理
                asyncio.create_task(
                    _handle_ws_chat(
                        websocket, student_id, course_id,
                        session_id, message,
                    )
                )

            elif msg_type == "ping":
                await websocket.send_json({"type": "pong"})

    except WebSocketDisconnect:
        ws_manager.disconnect(student_id, connection_id)
        if progress_task:
            progress_task.cancel()
    except Exception as e:
        logger.error(f"[WS] 错误: {e}")
        ws_manager.disconnect(student_id, connection_id)


async def _listen_task_progress(
    websocket: WebSocket, student_id: str, task_id: str
):
    """监听任务进度并通过 WebSocket 推送"""
    pubsub = await cache_service.subscribe_task(task_id)

    try:
        async for message in pubsub.listen():
            if message["type"] == "message":
                data = json.loads(message["data"])
                await websocket.send_json({
                    "type": "task_progress",
                    "task_id": task_id,
                    "data": data,
                })

                if data.get("status") in ("completed", "failed"):
                    break
    except asyncio.CancelledError:
        pass
    except Exception as e:
        logger.error(f"[WS] 进度监听错误: {e}")
    finally:
        await pubsub.unsubscribe()
```

```python
                        "task_id": task_id,
                        "progress": progress,
                        "current_step": current_step,
                        "completed_steps": completed_steps,

                        error_message = :error_message,
                        completed_at = NOW()

                text("SELECT * FROM generation_tasks WHERE id = :task_id"),
                {"task_id": task_id},
            )
            row = result.mappings().first()
            if row:
                return dict(row)

        return None
```

```python
三层防护:

    # 常见虚构模式
    FABRICATED_PATTERNS = [
        r"根据\S{2,10}教授\d{4}年的研究",    # 虚构的学者/年份
        r"发表在\S{2,10}期刊\d{4}年",         # 虚构的期刊
        r"实验表明成功率达到\d{2,3}%",         # 虚构的精确数据
        r"在\d{4}年获得\S{2,10}奖",           # 虚构的奖项
        r"根据\S{2,10}公司的\d{4}年报告",      # 虚构的报告
    ]

    async def check(
        self,
        generated_text: str,
        source_chunks: list[dict],
        check_level: str = "standard",
    ) -> HallucinationCheck:
        """
        执行防幻觉检查
        check_level: basic / standard / strict
        """
        # 第一层: 模式匹配检测虚构内容
        fabricated_refs = self._detect_fabricated_references(generated_text)

        # 第二层: 来源验证
        source_check = await self._verify_against_sources(
            generated_text, source_chunks
        )

        # 第三层: LLM 语义验证 (strict 模式)
        semantic_check = None
        if check_level == "strict":
            semantic_check = await self._semantic_verification(
                generated_text, source_chunks
            )

        # 综合判断
        has_hallucination = (
            len(fabricated_refs) > 0
            or source_check["unsupported_ratio"] > 0.3
            or (semantic_check and semantic_check.get("has_hallucination", False))
        )

        # 生成修正内容
        correction = ""
        safe_content = generated_text

        if has_hallucination:
            correction = await self._generate_correction(
                generated_text, fabricated_refs, source_check
            )
            safe_content = await self._generate_safe_version(
                generated_text, source_chunks
            )

        return HallucinationCheck(
            has_hallucination=has_hallucination,
            confidence=source_check.get("confidence", 0.5),
            suspicious_claims=source_check.get("unsupported", []),
            fabricated_references=fabricated_refs,
            correction=correction,
            safe_content=safe_content,
        )

    def _detect_fabricated_references(self, text: str) -> list[str]:
        """检测可能虚构的引用"""
        fabricated = []

        for pattern in self.FABRICATED_PATTERNS:
            matches = re.findall(pattern, text)
            fabricated.extend(matches)

        return fabricated

    async def _verify_against_sources(
        self, text: str, sources: list[dict]
    ) -> dict:
        """验证生成内容是否基于来源"""
        if not sources:
            return {
                "supported": [],
                "unsupported": [text[:100]],
                "unsupported_ratio": 1.0,
                "confidence": 0.1,
            }

        sources_text = "\n\n".join(
            f"[来源{i+1}] {s.get('content', '')[:400]}"
            for i, s in enumerate(sources[:5])
        )

        prompt = f"""请分析以下生成内容，判断哪些陈述有来源支持，哪些没有。

生成内容:
{text[:2000]}

参考资料:
{sources_text}

请返回 JSON:
{{
    "supported_claims": ["有来源支持的陈述1", "..."],
    "unsupported_claims": ["无来源支持的陈述1", "..."],
    "accuracy_score": 0.0~1.0
}}"""

        messages = [
            SparkMessage(role="system", content="你是内容审核助手。只返回JSON。"),
            SparkMessage(role="user", content=prompt),
        ]

        try:
            response = await spark_client.chat(messages, temperature=0.1, max_tokens=1000)

            import json

            match = re.search(r"\{.*\}", response.content, re.DOTALL)
            if match:
                result = json.loads(match.group())
                supported = result.get("supported_claims", [])
                unsupported = result.get("unsupported_claims", [])
                total = len(supported) + len(unsupported)

                return {
                    "supported": supported,
                    "unsupported": unsupported,
                    "unsupported_ratio": len(unsupported) / max(total, 1),
                    "confidence": result.get("accuracy_score", 0.5),
                }
        except Exception as e:
            logger.error(f"来源验证失败: {e}")

        return {
            "supported": [],
            "unsupported": [],
            "unsupported_ratio": 0.0,
            "confidence": 0.5,
        }

    async def _semantic_verification(
        self, text: str, sources: list[dict]
    ) -> dict:
        """LLM 语义级别的深度验证"""
        prompt = f"""请深度分析以下内容的准确性:
1. 是否包含事实性错误?
2. 是否有逻辑矛盾?
3. 是否有数据/数字可能被编造?
4. 是否有虚构的引用/来源?

内容:
{text[:1500]}

参考来源 (如有):
{str(sources)[:500]}

返回 JSON:
{{"has_hallucination": true/false, "issues": ["问题1", "..."], "severity": "low/medium/high"}}"""

        messages = [
            SparkMessage(role="system", content="你是严格的学术内容审核员。只返回JSON。"),
            SparkMessage(role="user", content=prompt),
        ]

        try:
            response = await spark_client.chat(messages, temperature=0.1, max_tokens=500)

            import json

            match = re.search(r"\{.*\}", response.content, re.DOTALL)
            if match:
                return json.loads(match.group())
        except Exception as e:
            logger.error(f"语义验证失败: {e}")

        return {"has_hallucination": False, "issues": [], "severity": "low"}

    async def _generate_correction(
        self,
        text: str,
        fabricated_refs: list[str],
        source_check: dict,
    ) -> str:
        """生成修正说明"""
        issues = []
        if fabricated_refs:
            issues.append(f"发现可能虚构的引用: {'; '.join(fabricated_refs[:3])}")
        if source_check.get("unsupported"):
            issues.append(
                f"以下内容无法从来源验证: {'; '.join(source_check['unsupported'][:3])}"
            )

        return "⚠️ 内容审核提示:\n" + "\n".join(f"- {issue}" for issue in issues)

    async def _generate_safe_version(
        self, text: str, sources: list[dict]
    ) -> str:
        """生成安全版本 (移除/标注不确定内容)"""
        sources_text = "\n".join(s.get("content", "")[:200] for s in sources[:3])

        prompt = f"""请基于参考资料，生成一个安全版本的回答。
要求:
1. 移除任何无法从参考资料验证的内容
2. 用 [待验证] 标注不确定的部分
3. 保留有来源支持的内容

原始内容:
{text[:1500]}

参考资料:
{sources_text}

只返回安全版本的文本。"""

        messages = [
            SparkMessage(role="system", content="你是内容安全助手。"),
            SparkMessage(role="user", content=prompt),
        ]

        try:
            response = await spark_client.chat(messages, temperature=0.2, max_tokens=2000)
            return response.content if response.content else text
        except Exception:
            return text
```

```python
# backend/app/services/content_safety.py
"""
内容安全服务
功能: 敏感词过滤 + LLM 语义安全检查
"""

import re
from dataclasses import dataclass

from loguru import logger

from app.services.spark_client import spark_client, SparkMessage


@dataclass
class SafetyResult:
    is_safe: bool
    risk_level: str          # low / medium / high / blocked
    violations: list[str]
    filtered_content: str


class ContentSafetyService:
    """内容安全服务"""

    # 敏感词库 (简化版, 实际项目应使用更完整的词库)
    SENSITIVE_KEYWORDS = {
        "political": ["政治敏感词1", "政治敏感词2"],
        "violent": ["暴力", "血腥"],
        "illegal": ["违法", "犯罪"],
        "discrimination": ["歧视"],
        "privacy": ["身份证号", "银行卡号", "密码"],
    }

    # 教育场景豁免词
    EDUCATION_WHITELIST = [
        "算法", "模型", "训练", "测试", "评估",
        "攻击", "防御", "对抗", "破解",  # 安全学术语境
    ]

    async def check(self, content: str) -> SafetyResult:
        """检查内容安全性"""
        # 第一层: 关键词过滤
        keyword_violations = self._keyword_check(content)

        # 第二层: LLM 语义安全检查
        semantic_check = await self._semantic_safety_check(content)

        # 综合判断
        violations = keyword_violations + semantic_check.get("violations", [])
        risk_level = self._assess_risk(violations, semantic_check)

        is_safe = risk_level in ("low", "medium")
        filtered_content = self._filter_content(content, keyword_violations) if keyword_violations else content

        return SafetyResult(
            is_safe=is_safe,
            risk_level=risk_level,
            violations=violations,
            filtered_content=filtered_content,
        )

    def _keyword_check(self, content: str) -> list[str]:
        """关键词过滤"""
        violations = []

        for category, keywords in self.SENSITIVE_KEYWORDS.items():
            for keyword in keywords:
                if keyword in content:
                    # 检查是否在教育白名单中
                    if not self._is_education_context(content, keyword):
                        violations.append(f"[{category}] 包含敏感词: {keyword}")

        return violations

    def _is_education_context(self, content: str, keyword: str) -> bool:
        """判断是否在教育语境中"""
        # 简化实现: 检查附近是否有教育相关词
        idx = content.find(keyword)
        if idx == -1:
            return False

        context = content[max(0, idx - 50) : idx + 50]
        return any(word in context for word in self.EDUCATION_WHITELIST)

    async def _semantic_safety_check(self, content: str) -> dict:
        """LLM 语义安全检查"""
        prompt = f"""请检查以下内容的安全性:
1. 是否包含不当内容 (暴力、色情、歧视等)?
2. 是否包含有害指导 (违法、危险行为等)?
3. 是否包含个人隐私信息?

内容:
{content[:1000]}

返回 JSON:
{{"is_safe": true/false, "violations": ["问题1", "..."], "risk_level": "low/medium/high"}}"""

        messages = [
            SparkMessage(role="system", content="你是内容安全审核员。只返回JSON。"),
            SparkMessage(role="user", content=prompt),
        ]

        try:
            import re as _re
            import json

            response = await spark_client.chat(messages, temperature=0.1, max_tokens=300)

            match = _re.search(r"\{.*\}", response.content, _re.DOTALL)
            if match:
                return json.loads(match.group())
        except Exception as e:
            logger.error(f"语义安全检查失败: {e}")

        return {"is_safe": True, "violations": [], "risk_level": "low"}

    def _assess_risk(self, violations: list[str], semantic_check: dict) -> str:
        """评估整体风险等级"""
        if not violations and semantic_check.get("is_safe", True):
            return "low"

        semantic_risk = semantic_check.get("risk_level", "low")

        if semantic_risk == "high" or len(violations) > 3:
            return "blocked"
        elif semantic_risk == "medium" or len(violations) > 1:
            return "medium"
        else:
            return "low"

    def _filter_content(self, content: str, violations: list[str]) -> str:
        """过滤敏感内容"""
        filtered = content

        for category, keywords in self.SENSITIVE_KEYWORDS.items():
            for keyword in keywords:
                if keyword in filtered and not self._is_education_context(filtered, keyword):
                    filtered = filtered.replace(keyword, "***")

        return filtered
```

```tsx
// src/components/chat/ChatWindow.tsx
"use client";

import React, { useState, useRef, useEffect, useCallback } from "react";
import { Send, Loader2, BookOpen, Map, Code, Volume2 } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { MessageBubble } from "./MessageBubble";
import { ResourceCard } from "./ResourceCard";
import { useChatStore } from "@/stores/useChatStore";
import type { ChatMessage, Resource, TaskProgress, SSEEvent } from "@/types";

interface ChatWindowProps {
  studentId: string;
  courseId: string;
  sessionId?: string;
}

export function ChatWindow({
  studentId,
  courseId,
  sessionId = "default",
}: ChatWindowProps) {
  const [input, setInput] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const [progress, setProgress] = useState<TaskProgress | null>(null);
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const eventSourceRef = useRef<EventSource | null>(null);

  const { messages, addMessage, updateLastMessage, addResource } =
    useChatStore();

  // 自动滚动到底部
  const scrollToBottom = useCallback(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, []);

  useEffect(() => {
    scrollToBottom();
  }, [messages, scrollToBottom]);

  // 清理 EventSource
  useEffect(() => {
    return () => {
      eventSourceRef.current?.close();
    };
  }, []);

  const handleSend = async () => {
    if (!input.trim() || isLoading) return;

    const userMessage: ChatMessage = {
      id: Date.now().toString(),
      role: "user",
      content: input.trim(),
      timestamp: new Date().toISOString(),
    };

    addMessage(userMessage);
    setInput("");
    setIsLoading(true);
    setProgress(null);

    // 创建空的 assistant 消息用于流式填充
    const assistantMessage: ChatMessage = {
      id: (Date.now() + 1).toString(),
      role: "assistant",
      content: "",
      resources: [],
      citations: [],
      timestamp: new Date().toISOString(),
    };
    addMessage(assistantMessage);

    try {
      // 使用 SSE 流式请求
      const params = new URLSearchParams({
        message: userMessage.content,
        course_id: courseId,
        session_id: sessionId,
      });

      const eventSource = new EventSource(
        `/api/v1/stream/chat?${params.toString()}`
      );
      eventSourceRef.current = eventSource;

      eventSource.addEventListener("start", (e) => {
        const data = JSON.parse(e.data);
        setProgress({
          task_id: data.task_id,
          status: "running",
          progress: 0.1,
          current_step: "开始处理...",
        });
      });

      eventSource.addEventListener("progress", (e) => {
        const data = JSON.parse(e.data);
        setProgress(data);
      });

      eventSource.addEventListener("resource", (e) => {
        const resource: Resource = JSON.parse(e.data);
        addResource(resource);
      });

      eventSource.addEventListener("message", (e) => {
        const data = JSON.parse(e.data);
        updateLastMessage({
          content: data.content,
          citations: data.citations || [],
        });
      });

      eventSource.addEventListener("done", () => {
        setIsLoading(false);
        setProgress(null);
        eventSource.close();
      });

      eventSource.addEventListener("error", (e) => {
        console.error("SSE error:", e);
        updateLastMessage({
          content: "抱歉，处理过程中出现错误，请重试。",
        });
        setIsLoading(false);
        setProgress(null);
        eventSource.close();
      });

      eventSource.onerror = () => {
        setIsLoading(false);
        setProgress(null);
        eventSource.close();
      };
    } catch (error) {
      console.error("Chat error:", error);
      updateLastMessage({
        content: "抱歉，连接出现问题，请重试。",
      });
      setIsLoading(false);
    }
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  return (
    <div className="flex flex-col h-full bg-gradient-to-b from-slate-50 to-white rounded-xl border border-slate-200 shadow-lg">
      {/* Header */}
      <div className="flex items-center gap-3 px-6 py-4 border-b border-slate-200 bg-white rounded-t-xl">
        <div className="w-10 h-10 rounded-full bg-gradient-to-br from-blue-500 to-indigo-600 flex items-center justify-center">
          <BookOpen className="w-5 h-5 text-white" />
        </div>
        <div>
          <h2 className="text-lg font-semibold text-slate-800">智枢助手</h2>
          <p className="text-xs text-slate-500">AI 导论课程 · 智能辅导</p>
        </div>
        {isLoading && (
          <div className="ml-auto flex items-center gap-2 text-sm text-blue-600">
            <Loader2 className="w-4 h-4 animate-spin" />
            <span>{progress?.current_step || "思考中..."}</span>
          </div>
        )}
      </div>

      {/* Progress Bar */}
      {progress && progress.status === "running" && (
        <div className="px-6 py-2 bg-blue-50 border-b border-blue-100">
```

```tsx
"use client";

import React, { useEffect, useRef, useState } from "react";
import { ZoomIn, ZoomOut, Download, Maximize2 } from "lucide-react";
import { Button } from "@/components/ui/button";

interface MindMapCardProps {

export function MindMapCard({ code, className = "" }: MindMapCardProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [scale, setScale] = useState(1);

  useEffect(() => {
    renderMermaid();
  }, [code]);

  const renderMermaid = async () => {
    if (!containerRef.current || !code) return;

    setIsLoading(true);
    setError(null);

    try {
      const mermaid = (await import("mermaid")).default;

      mermaid.initialize({
        startOnLoad: false,
        theme: "default",
        securityLevel: "loose",
        mindmap: {
          padding: 20,
        },
        themeVariables: {
          primaryColor: "#3b82f6",
          primaryTextColor: "#1e293b",
          primaryBorderColor: "#93c5fd",
          lineColor: "#94a3b8",
          secondaryColor: "#e0f2fe",
          tertiaryColor: "#f0f9ff",
        },
      });

      const id = `mindmap-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;

      // 清理容器
      containerRef.current.innerHTML = "";

      const { svg } = await mermaid.render(id, code);
      containerRef.current.innerHTML = svg;

      setIsLoading(false);
    } catch (err: any) {
      console.error("Mermaid render error:", err);
      setError(err.message || "渲染失败");
      setIsLoading(false);
    }
  };

  const handleZoomIn = () => setScale((s) => Math.min(s + 0.2, 2));
  const handleZoomOut = () => setScale((s) => Math.max(s - 0.2, 0.5));

  const handleDownload = () => {
    if (!containerRef.current) return;

    const svgElement = containerRef.current.querySelector("svg");
    if (!svgElement) return;

    const svgData = new XMLSerializer().serializeToString(svgElement);
    const blob = new Blob([svgData], { type: "image/svg+xml" });
    const url = URL.createObjectURL(blob);

    const a = document.createElement("a");
    a.href = url;
    a.download = "mindmap.svg";
    a.click();

    URL.revokeObjectURL(url);
  };

  return (
    <div
      className={`relative bg-white rounded-lg border border-slate-200 overflow-hidden ${className}`}
    >
      {/* 工具栏 */}
      <div className="absolute top-2 right-2 flex gap-1 z-10">
        <Button
          size="sm"
          variant="ghost"
          onClick={handleZoomIn}
          className="w-7 h-7 p-0 bg-white/80 backdrop-blur"
        >
          <ZoomIn className="w-3.5 h-3.5" />
        </Button>
        <Button
          size="sm"
          variant="ghost"
          onClick={handleZoomOut}
          className="w-7 h-7 p-0 bg-white/80 backdrop-blur"
        >
          <ZoomOut className="w-3.5 h-3.5" />
        </Button>
        <Button
          size="sm"
          variant="ghost"
          onClick={handleDownload}
          className="w-7 h-7 p-0 bg-white/80 backdrop-blur"
        >
          <Download className="w-3.5 h-3.5" />
        </Button>
      </div>

      {/* 内容区 */}
      <div className="p-4 overflow-auto max-h-[400px]">
        {isLoading && (
          <div className="flex items-center justify-center h-32 text-sm text-slate-400">
```

```typescript
  setCurrentTaskId: (taskId: string | null) => void;
  clearChat: () => void;
}

export const useChatStore = create<ChatState>((set, get) => ({
  messages: [],
  resources: [],
  isLoading: false,
  currentTaskId: null,

  addMessage: (msg) =>
    set((state) => ({ messages: [...state.messages, msg] })),

  updateLastMessage: (updates) =>
    set((state) => {
      const messages = [...state.messages];
      const lastIdx = messages.length - 1;
      if (lastIdx >= 0 && messages[lastIdx].role === "assistant") {
        messages[lastIdx] = { ...messages[lastIdx], ...updates };
      }
      return { messages };
    }),

  addResource: (resource) =>
    set((state) => ({ resources: [...state.resources, resource] })),

  setLoading: (loading) => set({ isLoading: loading }),

  setCurrentTaskId: (taskId) => set({ currentTaskId: taskId }),

  clearChat: () =>
    set({ messages: [], resources: [], currentTaskId: null }),
}));
```

```typescript
    const headers: Record<string, string> = {
      "Content-Type": "application/json",
      ...(options.headers as Record<string, string>),
    };

    if (this.token) {
      headers["Authorization"] = `Bearer ${this.token}`;
    }

    const response = await fetch(`${BASE_URL}${endpoint}`, {
      ...options,
      headers,
    });

    if (!response.ok) {
      const error = await response.json().catch(() => ({}));
      throw new Error(error.detail || `API Error: ${response.status}`);
    }

    return response.json();
  }

  // --- 画像 API ---
  async buildProfile(data: {
    student_id: string;
    course_id: string;
    answers: Record<string, any>;
  }) {
    return this.request<any>("/profile/build", {
      method: "POST",
      body: JSON.stringify(data),
    });
  }

  async getProfile(studentId: string, courseId: string) {
    return this.request<any>(`/profile/${studentId}/${courseId}`);
  }

  async updateProfile(studentId: string, courseId: string, data: any) {
    return this.request<any>(`/profile/${studentId}/${courseId}`, {
      method: "PUT",
      body: JSON.stringify(data),
    });
  }

  // --- 资源 API ---
  async generateResource(data: {
    student_id: string;
    course_id: string;
    topic: string;
    type: string;
  }) {
    return this.request<any>("/resources/generate", {

    });
  }

  async listResources(studentId: string, courseId: string, type?: string) {
    const params = new URLSearchParams({ student_id: studentId, course_id: courseId });
    if (type) params.set("type", type);
    return this.request<any[]>(`/resources?${params.toString()}`);
  }

  // --- 路径 API ---
  async generatePath(studentId: string, courseId: string) {
    return this.request<any>("/path/generate", {
```

```python
@router.post("/build", response_model=ProfileResponse)
async def build_profile(
    request: ProfileBuildRequest,
    db: AsyncSession = Depends(get_db),
):
    """构建学习画像 (从问卷)"""
    # 检查是否已有画像
    existing = await db.execute(
        text("""
            SELECT id FROM student_profiles
            WHERE student_id = :sid AND course_id = :cid
        """),
        {"sid": request.student_id, "cid": request.course_id},
    )
    if existing.fetchone():
        raise HTTPException(status_code=409, detail="画像已存在，请使用更新接口")

    # 调用 Profile Agent 构建画像
    state = {
        "student_id": request.student_id,
        "course_id": request.course_id,
        "user_message": str(request.answers),
        "student_profile": {},
        "intent_params": {"action": "build"},
    }

    result = await profile_agent.execute(state)

    # 存储画像
    dimensions = result.get("dimensions", {})
    confidence = result.get("confidence", 0.5)

    import json

    insert_result = await db.execute(
        text("""
            INSERT INTO student_profiles
                (student_id, course_id, dimensions, confidence, source, raw_answers)
            VALUES
                (:sid, :cid, :dims, :conf, 'questionnaire', :answers)
            RETURNING id, version
        """),
        {
            "sid": request.student_id,
            "cid": request.course_id,
            "dims": json.dumps(dimensions, ensure_ascii=False),
            "conf": confidence,
            "answers": json.dumps(request.answers, ensure_ascii=False),
        },
    )
    row = insert_result.fetchone()
    await db.commit()

    # 更新缓存
    await cache_service.set_profile(
        request.student_id,
        request.course_id,
        {"dimensions": dimensions, "confidence": confidence},
    )

    return ProfileResponse(
        id=str(row[0]),
        student_id=request.student_id,
        course_id=request.course_id,
        dimensions=dimensions,
        confidence=confidence,
        version=row[1],
    )


@router.get("/{student_id}/{course_id}", response_model=ProfileResponse)
async def get_profile(
    student_id: str,
    course_id: str,
    db: AsyncSession = Depends(get_db),
):
    """获取学习画像"""
    # 先查缓存
    cached = await cache_service.get_profile(student_id, course_id)
    if cached:
        return ProfileResponse(
            id=cached.get("id", ""),
            student_id=student_id,
            course_id=course_id,
            dimensions=cached.get("dimensions", {}),
            confidence=cached.get("confidence", 0),
            version=cached.get("version", 1),
        )

    # 查数据库
    result = await db.execute(
        text("""
            SELECT id, dimensions, confidence, version
            FROM student_profiles
            WHERE student_id = :sid AND course_id = :cid
        """),
        {"sid": student_id, "cid": course_id},
    )
    row = result.mappings().first()

    if not row:
        raise HTTPException(status_code=404, detail="画像不存在")

    profile_data = {
        "id": str(row["id"]),
        "dimensions": dict(row["dimensions"]) if row["dimensions"] else {},
        "confidence": float(row["confidence"]) if row["confidence"] else 0,
        "version": row["version"],
    }

    # 写入缓存
    await cache_service.set_profile(student_id, course_id, profile_data)

    return ProfileResponse(
        student_id=student_id,
        course_id=course_id,
        **profile_data,
    )


@router.put("/{student_id}/{course_id}")

    student_id: str,
    course_id: str,
    request: ProfileUpdateRequest,
    db: AsyncSession = Depends(get_db),

    """更新学习画像"""
    state = {
        "student_id": student_id,
        "course_id": course_id,
        "user_message": str(request.data),
        "student_profile": {},
        "intent_params": {"action": request.action, **request.data},
    }

    # 获取当前画像
    result = await db.execute(
        text("""
            SELECT dimensions, confidence FROM student_profiles
            WHERE student_id = :sid AND course_id = :cid
        """),
        {"sid": student_id, "cid": course_id},
    )
    row = result.mappings().first()
    if row:
        state["student_profile"] = {
            "dimensions": dict(row["dimensions"]) if row["dimensions"] else {},
            "confidence": float(row["confidence"]) if row["confidence"] else 0,
        }

    update_result = await profile_agent.execute(state)

    # 更新数据库
    import json

    await db.execute(
        text("""
            UPDATE student_profiles
            SET dimensions = :dims,
                confidence = :conf,
                version = version + 1,
                updated_at = NOW()
            WHERE student_id = :sid AND course_id = :cid
        """),

            "cid": course_id,
            "dims": json.dumps(update_result.get("dimensions", {}), ensure_ascii=False),
            "conf": update_result.get("confidence", 0),
        },
    )
    await db.commit()
```

```python
@router.post("/generate")
async def generate_resource(
    request: ResourceGenerateRequest,
    db: AsyncSession = Depends(get_db),

    """生成学习资源 (异步任务)"""
    task_id = str(uuid4())

    # 创建任务记录
    await task_tracker.create_task(
        task_id=task_id,
        task_type="resource_gen",
        student_id=request.student_id,
        total_steps=3,
    )

    # 异步执行 (实际应使用 Celery)
    import asyncio

    async def _execute():
        try:
            await task_tracker.update_progress(
                task_id, 0.1, "正在分析需求...", 1
            )

            result = await master_agent.run({
                "student_id": request.student_id,
                "course_id": request.course_id,
                "user_message": f"请为我生成关于 {request.topic} 的学习资源",
                "intent": request.resource_type,
                "intent_params": {"topic": request.topic},
                "task_id": task_id,
            })

            # 存储生成的资源
            await _save_resources(db, request.student_id, request.course_id, result)

            await task_tracker.complete_task(task_id, {
                "resources": result.get("resources", []),
                "response": result.get("final_response", ""),
            })

        except Exception as e:
            logger.error(f"资源生成失败: {e}")
            await task_tracker.fail_task(task_id, str(e))

    asyncio.create_task(_execute())

    return {"task_id": task_id, "status": "pending"}


@router.get("/")
async def list_resources(
    student_id: str = Query(...),
    course_id: str = Query(...),
    resource_type: Optional[str] = Query(None),
    limit: int = Query(20, le=100),
    offset: int = Query(0),
    db: AsyncSession = Depends(get_db),
):
    """查询资源列表"""
    query = """
        SELECT id, type, title, content, mermaid_code, exercises,
               code_snippets, audio_url, source_refs, is_verified, created_at
        FROM resources
        WHERE student_id = :sid AND course_id = :cid
    """
    params = {"sid": student_id, "cid": course_id, "limit": limit, "offset": offset}

    if resource_type:
        query += " AND type = :rtype"
        params["rtype"] = resource_type

    query += " ORDER BY created_at DESC LIMIT :limit OFFSET :offset"

    result = await db.execute(text(query), params)
    rows = result.mappings().all()

    return [
        ResourceResponse(
            id=str(row["id"]),
            type=row["type"],
            title=row["title"],
            content=row["content"],
            mermaid_code=row["mermaid_code"],
            exercises=row["exercises"],
            code_snippets=row["code_snippets"],
            audio_url=row["audio_url"],
            source_refs=row["source_refs"],
            is_verified=row["is_verified"],
            created_at=str(row["created_at"]),
        )
        for row in rows
    ]


async def _save_resources(db, student_id, course_id, result):
    """保存生成的资源到数据库"""
    import json

    for resource in result.get("resources", []):
        await db.execute(
            text("""
                INSERT INTO resources
                    (student_id, course_id, type, title, content,
                     mermaid_code, exercises, code_snippets, source_refs, is_verified)
                VALUES
                    (:sid, :cid, :type, :title, :content,
                     :mermaid, :exercises, :code, :refs, :verified)
            """),
            {
                "sid": student_id,
                "cid": course_id,
                "type": resource.get("type", "explanation"),
                "title": resource.get("title", ""),
                "content": resource.get("content"),
                "mermaid": resource.get("mermaid_code"),
                "exercises": json.dumps(resource.get("exercises"), ensure_ascii=False) if resource.get("exercises") else None,
                "code": json.dumps(resource.get("code_snippets"), ensure_ascii=False) if resource.get("code_snippets") else None,
                "refs": json.dumps(resource.get("source_refs", []), ensure_ascii=False),
                "verified": resource.get("is_verified", False),
            },
        )
    await db.commit()
```

## 14. 开发计划

### 策略说明

采用 **Vertical Slice（垂直切片）** 策略：D1 立起 Master Agent + Document Agent 的端到端链路，之后每天叠加一个子 Agent。每新增一个 Agent 只需 `add_node` + 一行路由，骨架已验证过，零风险。这样 D1 就能跑通全链路，而不是 D14 才首次联调。

### 15 天详细任务分解

| 天数 | 日期 | 任务 | 产出 | 负责人 |
|------|------|------|------|--------|
