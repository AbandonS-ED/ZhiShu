## 1. 概述

### 1.1 为什么选择 Zustand

- **轻量级**: 体积小，性能好
- **简单易用**: API 简洁，学习成本低
- **TypeScript 友好**: 完美的类型支持
- **无 Provider**: 不需要包裹 Provider 组件
- **支持中间件**: 支持持久化、日志等中间件

### 1.2 状态划分

| Store | 职责 | 数据 |
|-------|------|------|
| useAppStore | 全局状态 | 学生信息、课程信息、侧边栏状态 |
| useChatStore | 对话状态 | 消息列表、资源列表、加载状态 |

## 2. 全局状态 (useAppStore)

### 2.1 Store 定义

**文件位置**: `src/stores/useAppStore.ts`

```typescript
import { create } from "zustand";
import { persist, createJSONStorage } from "zustand/middleware";
import type { Student, StudentProfile, LearningPath } from "@/types";

interface AppState {
  // 学生信息
  student: Student | null;
  setStudent: (student: Student | null) => void;

  // 画像信息
  profile: StudentProfile | null;
  setProfile: (profile: StudentProfile | null) => void;

  // 学习路径
  currentPath: LearningPath | null;
  setCurrentPath: (path: LearningPath | null) => void;

  // 课程信息
  courseId: string;
  setCourseId: (courseId: string) => void;

  // 侧边栏状态
  sidebarOpen: boolean;
  toggleSidebar: () => void;
  setSidebarOpen: (open: boolean) => void;

  // 主题
  theme: "light" | "dark" | "system";
  setTheme: (theme: "light" | "dark" | "system") => void;

  // 重置状态
  reset: () => void;
}

const initialState = {
  student: null,
  profile: null,
  currentPath: null,
  courseId: "",
  sidebarOpen: true,
  theme: "system" as const,
};

export const useAppStore = create<AppState>()(
  persist(
    (set) => ({
      ...initialState,

      setStudent: (student) => set({ student }),

      setProfile: (profile) => set({ profile }),

      setCurrentPath: (path) => set({ currentPath: path }),

      setCourseId: (courseId) => set({ courseId }),

      toggleSidebar: () => set((state) => ({ sidebarOpen: !state.sidebarOpen })),

      setSidebarOpen: (open) => set({ sidebarOpen: open }),

      setTheme: (theme) => set({ theme }),

      reset: () => set(initialState),
    }),
    {
      name: "zhishu-app-storage",
      storage: createJSONStorage(() => localStorage),
      partialize: (state) => ({
        // 只持久化必要字段
        courseId: state.courseId,
        sidebarOpen: state.sidebarOpen,
        theme: state.theme,
      }),
    }
  )
);
```

### 2.2 使用示例

```tsx
"use client";

import React from "react";
import { useAppStore } from "@/stores/useAppStore";

export function UserProfile() {
  const { student, setStudent } = useAppStore();

  return (
    <div>
      {student ? (
        <div>
          <p>姓名: {student.name}</p>

  return (
    <button onClick={toggleSidebar}>
      {sidebarOpen ? "收起侧边栏" : "展开侧边栏"}
    </button>
  );
}
```

## 3. 对话状态 (useChatStore)

### 3.1 Store 定义

**文件位置**: `src/stores/useChatStore.ts`

```typescript
import { create } from "zustand";
import type { ChatMessage, Resource, TaskProgress } from "@/types";

interface ChatState {
  // 消息列表
  messages: ChatMessage[];
  addMessage: (msg: ChatMessage) => void;
  updateMessage: (id: string, updates: Partial<ChatMessage>) => void;
  updateLastMessage: (updates: Partial<ChatMessage>) => void;
  clearMessages: () => void;

  // 资源列表
  resources: Resource[];
  addResource: (resource: Resource) => void;
  clearResources: () => void;

  // 加载状态
  isLoading: boolean;
  setLoading: (loading: boolean) => void;

  // 当前任务
  currentTaskId: string | null;
  setCurrentTaskId: (taskId: string | null) => void;

  // 任务进度
  progress: TaskProgress | null;
  setProgress: (progress: TaskProgress | null) => void;

  // 会话 ID
  sessionId: string;
  setSessionId: (sessionId: string) => void;

  // 清空所有状态
  clearChat: () => void;
}

export const useChatStore = create<ChatState>((set, get) => ({
  // 消息列表
  messages: [],
  addMessage: (msg) =>
    set((state) => ({ messages: [...state.messages, msg] })),

  updateMessage: (id, updates) =>
    set((state) => ({
      messages: state.messages.map((msg) =>
        msg.id === id ? { ...msg, ...updates } : msg
      ),
    })),

  updateLastMessage: (updates) =>
    set((state) => {
      const messages = [...state.messages];
      const lastIdx = messages.length - 1;
      if (lastIdx >= 0 && messages[lastIdx].role === "assistant") {
        messages[lastIdx] = { ...messages[lastIdx], ...updates };
      }
      return { messages };
    }),

  clearMessages: () => set({ messages: [] }),

  // 资源列表
  resources: [],
  addResource: (resource) =>
    set((state) => ({ resources: [...state.resources, resource] })),

  clearResources: () => set({ resources: [] }),

  // 加载状态
  isLoading: false,
  setLoading: (loading) => set({ isLoading: loading }),

  // 当前任务
  currentTaskId: null,
  setCurrentTaskId: (taskId) => set({ currentTaskId: taskId }),

  // 任务进度
  progress: null,
  setProgress: (progress) => set({ progress }),

  // 会话 ID
  sessionId: "default",
  setSessionId: (sessionId) => set({ sessionId }),

  // 清空所有状态
  clearChat: () =>
    set({
      messages: [],
      resources: [],
      isLoading: false,
      currentTaskId: null,
      progress: null,
    }),
}));
```

### 3.2 使用示例

```tsx
"use client";

import React, { useState } from "react";
import { useChatStore } from "@/stores/useChatStore";
import { useSSE } from "@/hooks/useSSE";

export function ChatWindow() {
  const [input, setInput] = useState("");
  const {
    messages,
    resources,
    isLoading,
    progress,
    addMessage,
    updateLastMessage,
    addResource,
    setLoading,
    setProgress,
  } = useChatStore();

  const { connect, disconnect } = useSSE({
    url: "/api/v1/stream/chat",
    params: { message: input },
    onStart: (data) => {
      setProgress({
        task_id: data.task_id,
        status: "running",
        progress: 0.1,
        current_step: "开始处理...",
      });
    },
    onProgress: (data) => {
      setProgress(data);
    },
    onResource: (resource) => {
      addResource(resource);
    },
    onMessage: (data) => {
      updateLastMessage({
        content: data.content,
        citations: data.citations || [],
      });
    },
    onDone: () => {
      setLoading(false);
      setProgress(null);
    },
  });

  const handleSend = () => {
    if (!input.trim()) return;

    // 添加用户消息
    addMessage({
      id: Date.now().toString(),
      role: "user",
      content: input,
      timestamp: new Date().toISOString(),
    });

    // 添加助手消息占位
    addMessage({
      id: (Date.now() + 1).toString(),
      role: "assistant",
      content: "",
      timestamp: new Date().toISOString(),
    });

    // 设置加载状态
    setLoading(true);
    setInput("");

    // 连接 SSE
    connect();
  };

  return (
    <div className="flex flex-col h-full">
      {/* 消息列表 */}
      <div className="flex-1 overflow-y-auto p-4 space-y-4">
        {messages.map((msg) => (
          <div
            key={msg.id}
            className={`flex ${
              msg.role === "user" ? "justify-end" : "justify-start"
            }`}
          >
            <div
              className={`max-w-[70%] rounded-2xl px-4 py-2 ${
                msg.role === "user"
                  ? "bg-blue-500 text-white"
                  : "bg-slate-100 text-slate-800"
              }`}
            >
              {msg.content || (
                <span className="animate-pulse">思考中...</span>
              )}
            </div>
          </div>
        ))}
      </div>

      {/* 资源列表 */}
      {resources.length > 0 && (
        <div className="px-4 py-2 border-t border-slate-200">
          <div className="flex gap-2 overflow-x-auto">
            {resources.map((resource, i) => (
              <div
                key={i}
                className="px-3 py-1 bg-slate-100 rounded-full text-sm text-slate-600"
              >
                {resource.title}
              </div>
            ))}
          </div>
        </div>
      )}

      {/* 进度条 */}
      {progress && progress.status === "running" && (
        <div className="px-4 py-2 bg-blue-50 border-t border-blue-100">
          <div className="flex items-center justify-between text-xs text-blue-700 mb-1">
            <span>{progress.current_step}</span>
            <span>{Math.round(progress.progress * 100)}%</span>
          </div>
          <div className="w-full h-1.5 bg-blue-200 rounded-full overflow-hidden">
            <div
              className="h-full bg-blue-500 rounded-full transition-all duration-500"
              style={{ width: `${progress.progress * 100}%` }}
            />
          </div>
        </div>
      )}

      {/* 输入框 */}
      <div className="p-4 border-t border-slate-200">
        <div className="flex gap-2">
          <input
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => {
              if (e.key === "Enter" && !e.shiftKey) {
                e.preventDefault();
                handleSend();
              }
            }}
            placeholder="输入你的问题..."
            disabled={isLoading}
            className="flex-1 px-4 py-2 border border-slate-200 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
          />
          <button
            onClick={handleSend}
            disabled={isLoading || !input.trim()}
            className="px-4 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600 disabled:opacity-50 disabled:cursor-not-allowed"
          >
            {isLoading ? "处理中..." : "发送"}
          </button>
        </div>
      </div>
    </div>
  );
}
```

## 5. 状态选择器

### 5.1 性能优化

为了避免不必要的重渲染，使用选择器选择特定状态：

```typescript
// ❌ 不推荐：任何状态变化都会重渲染
const store = useAppStore();

// ✅ 推荐：只选择需要的状态
const student = useAppStore((state) => state.student);
const sidebarOpen = useAppStore((state) => state.sidebarOpen);

// ✅ 推荐：使用多个选择器
const student = useAppStore((state) => state.student);
const setStudent = useAppStore((state) => state.setStudent);
```

### 5.2 复杂选择器

```typescript
// 选择多个状态
const { student, profile, courseId } = useAppStore((state) => ({
  student: state.student,
  profile: state.profile,
  courseId: state.courseId,
}));

// 使用 shallow 比较 (避免对象引用变化导致的重渲染)
import { shallow } from "zustand/shallow";

const { student, profile } = useAppStore(
  (state) => ({
    student: state.student,
    profile: state.profile,
  }),
  shallow
);
```

## 6. 状态订阅

### 6.1 订阅状态变化

```typescript
import { useAppStore } from "@/stores/useAppStore";

// 订阅特定状态变化
const unsubscribe = useAppStore.subscribe(
  (state) => state.student,
  (student, prevStudent) => {
    console.log("Student changed:", student, prevStudent);
  },
  { equalityFn: shallow }
);

// 取消订阅
unsubscribe();
```

## 9. 最佳实践

### 9.1 状态设计原则

1. **单一职责**: 每个 Store 只管理相关状态
2. **扁平化**: 避免嵌套过深的状态结构
3. **不可变性**: 使用展开运算符更新状态
4. **最小化**: 只存储必要状态，派生状态使用计算

### 9.2 性能优化

1. **使用选择器**: 避免不必要的重渲染
2. **使用 shallow 比较**: 对象/数组状态使用 shallow
3. **避免内联函数**: 使用 useCallback 或在 Store 中定义
4. **合理使用持久化**: 只持久化必要字段

### 9.3 错误处理

1. **异步操作**: 使用 try-catch 处理错误
2. **状态回滚**: 错误时恢复到之前状态
3. **用户反馈**: 及时显示错误信息

### 9.4 TypeScript

1. **严格类型**: 为 State 和 Action 定义明确类型
2. **泛型支持**: 使用泛型增强类型推断
3. **类型导出**: 导出类型供其他文件使用

## 10. 常见问题

### 10.1 状态不更新

**问题**: 调用 setState 后状态不更新

**解决方案**:
```typescript
// ❌ 错误：直接修改对象
state.student.name = "new name";
set(state);

// ✅ 正确：创建新对象
set((state) => ({
  student: { ...state.student, name: "new name" },
}));
```

### 10.2 组件不重渲染

**问题**: 状态更新了但组件不重渲染

**解决方案**:
```typescript
// ❌ 错误：选择整个 store
const store = useAppStore();

// ✅ 正确：使用选择器
const student = useAppStore((state) => state.student);
```

### 10.3 持久化数据丢失

**问题**: 刷新页面后持久化数据丢失

**解决方案**:
1. 检查 localStorage 是否被禁用
2. 检查 partialize 配置
3. 检查 storage 配置
