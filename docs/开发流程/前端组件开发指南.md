## 2. 对话组件

### 2.1 ChatWindow.tsx

**文件位置**: `src/components/chat/ChatWindow.tsx`

**功能**: 对话窗口，支持 SSE 流式输出

**核心逻辑**:

```tsx
"use client";

import React, { useState, useRef, useEffect, useCallback } from "react";
import { Send, Loader2, BookOpen, Map, Code, Volume2 } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { MessageBubble } from "./MessageBubble";
import { ResourceCard } from "./ResourceCard";
import { useChatStore } from "@/stores/useChatStore";
import type { ChatMessage, Resource, TaskProgress } from "@/types";

interface ChatWindowProps {
  studentId: string;
  courseId: string;
  sessionId?: string;
}

export function ChatWindow({
  studentId,
  courseId,
  sessionId = "default",
}: ChatWindowProps) {
  const [input, setInput] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const [progress, setProgress] = useState<TaskProgress | null>(null);
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const eventSourceRef = useRef<EventSource | null>(null);

  const { messages, addMessage, updateLastMessage, addResource } =
    useChatStore();

  // 自动滚动到底部
  const scrollToBottom = useCallback(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, []);

  useEffect(() => {
    scrollToBottom();
  }, [messages, scrollToBottom]);

  // 清理 EventSource
  useEffect(() => {
    return () => {
      eventSourceRef.current?.close();
    };
  }, []);

  const handleSend = async () => {
    if (!input.trim() || isLoading) return;

    const userMessage: ChatMessage = {
      id: Date.now().toString(),
      role: "user",
      content: input.trim(),
      timestamp: new Date().toISOString(),
    };

    addMessage(userMessage);
    setInput("");
    setIsLoading(true);
    setProgress(null);

    // 创建空的 assistant 消息用于流式填充
    const assistantMessage: ChatMessage = {
      id: (Date.now() + 1).toString(),
      role: "assistant",
      content: "",
      resources: [],
      citations: [],
      timestamp: new Date().toISOString(),
    };
    addMessage(assistantMessage);

    try {
      // 使用 SSE 流式请求
      const params = new URLSearchParams({
        message: userMessage.content,
        course_id: courseId,
        session_id: sessionId,
      });

      const eventSource = new EventSource(
        `/api/v1/stream/chat?${params.toString()}`
      );
      eventSourceRef.current = eventSource;

      eventSource.addEventListener("start", (e) => {
        const data = JSON.parse(e.data);
        setProgress({
          task_id: data.task_id,
          status: "running",
          progress: 0.1,
          current_step: "开始处理...",
        });
      });

      eventSource.addEventListener("progress", (e) => {
        const data = JSON.parse(e.data);
        setProgress(data);
      });

      eventSource.addEventListener("resource", (e) => {
        const resource: Resource = JSON.parse(e.data);
        addResource(resource);
      });

      eventSource.addEventListener("message", (e) => {
        const data = JSON.parse(e.data);
        updateLastMessage({
          content: data.content,
          citations: data.citations || [],
        });
      });

      eventSource.addEventListener("done", () => {
        setIsLoading(false);
        setProgress(null);
        eventSource.close();
      });

      eventSource.addEventListener("error", (e) => {
        console.error("SSE error:", e);
        updateLastMessage({
          content: "抱歉，处理过程中出现错误，请重试。",
        });
        setIsLoading(false);
        setProgress(null);
        eventSource.close();
      });

      eventSource.onerror = () => {
        setIsLoading(false);
        setProgress(null);
        eventSource.close();
      };
    } catch (error) {
      console.error("Chat error:", error);
      updateLastMessage({
        content: "抱歉，连接出现问题，请重试。",
      });
      setIsLoading(false);
    }
  };

  return (
    <div className="flex flex-col h-full bg-gradient-to-b from-slate-50 to-white rounded-xl border border-slate-200 shadow-lg">
      {/* Header */}
      <div className="flex items-center gap-3 px-6 py-4 border-b border-slate-200 bg-white rounded-t-xl">
        <div className="w-10 h-10 rounded-full bg-gradient-to-br from-blue-500 to-indigo-600 flex items-center justify-center">
          <BookOpen className="w-5 h-5 text-white" />
        </div>
        <div>
          <h2 className="text-lg font-semibold text-slate-800">智枢助手</h2>
          <p className="text-xs text-slate-500">AI 导论课程 · 智能辅导</p>
        </div>
        {isLoading && (
          <div className="ml-auto flex items-center gap-2 text-sm text-blue-600">
            <Loader2 className="w-4 h-4 animate-spin" />
            <span>{progress?.current_step || "思考中..."}</span>
          </div>
        )}
      </div>

      {/* Progress Bar */}
      {progress && progress.status === "running" && (
        <div className="px-6 py-2 bg-blue-50 border-b border-blue-100">
          <div className="flex items-center justify-between text-xs text-blue-700 mb-1">
            <span>{progress.current_step}</span>
            <span>{Math.round(progress.progress * 100)}%</span>
          </div>
          <div className="w-full h-1.5 bg-blue-200 rounded-full overflow-hidden">
            <div
              className="h-full bg-gradient-to-r from-blue-500 to-indigo-500 rounded-full transition-all duration-500"
              style={{ width: `${progress.progress * 100}%` }}
            />
          </div>
        </div>
      )}

      {/* Messages */}
      <div className="flex-1 overflow-y-auto px-6 py-4 space-y-4">
        {messages.length === 0 && (
          <div className="flex flex-col items-center justify-center h-full text-center">
            {/* 空状态 */}
          </div>
        )}

        {messages.map((msg) => (
          <MessageBubble key={msg.id} message={msg} />
        ))}

        <div ref={messagesEndRef} />
      </div>

      {/* Resources */}
      {useChatStore.getState().resources.length > 0 && (
        <div className="px-6 py-3 border-t border-slate-100 bg-slate-50/50">
          <div className="flex gap-3 overflow-x-auto pb-2">
            {useChatStore
              .getState()
              .resources.slice(-5)
              .map((resource, i) => (
                <ResourceCard key={i} resource={resource} compact />
              ))}
          </div>
        </div>
      )}

      {/* Input */}
      <div className="px-6 py-4 border-t border-slate-200 bg-white rounded-b-xl">
        <div className="flex gap-3">
          <Input
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => {
              if (e.key === "Enter" && !e.shiftKey) {
                e.preventDefault();
                handleSend();
              }
            }}
            placeholder="输入你的问题..."
            disabled={isLoading}
            className="flex-1 bg-slate-50 border-slate-200 focus:bg-white focus:border-blue-400 focus:ring-blue-400"
          />
          <Button
            onClick={handleSend}
            disabled={!input.trim() || isLoading}
            className="bg-gradient-to-r from-blue-500 to-indigo-600 hover:from-blue-600 hover:to-indigo-700 text-white shadow-md"
          >
            {isLoading ? (
              <Loader2 className="w-4 h-4 animate-spin" />
            ) : (
              <Send className="w-4 h-4" />
            )}
          </Button>
        </div>
      </div>
    </div>
  );
}
```

**SSE 事件处理**:
- `start`: 任务开始，显示进度条
- `progress`: 更新进度信息
- `resource`: 接收生成的资源
- `message`: 接收流式消息内容
- `done`: 任务完成
- `error`: 错误处理

### 2.2 MessageBubble.tsx

**文件位置**: `src/components/chat/MessageBubble.tsx`

**功能**: 消息气泡，区分用户和助手消息

**实现要点**:
```tsx
"use client";

import React from "react";
import ReactMarkdown from "react-markdown";
import { Prism as SyntaxHighlighter } from "react-syntax-highlighter";
import { oneDark } from "react-syntax-highlighter/dist/esm/styles/prism";
import { User, Bot } from "lucide-react";
import { cn } from "@/lib/utils";
import type { ChatMessage } from "@/types";

interface MessageBubbleProps {
  message: ChatMessage;
}

export function MessageBubble({ message }: MessageBubbleProps) {
  const isUser = message.role === "user";

  return (
    <div className={cn("flex gap-3", isUser ? "justify-end" : "justify-start")}>
      {/* 头像 */}
      {!isUser && (
        <div className="w-8 h-8 rounded-full bg-gradient-to-br from-blue-500 to-indigo-600 flex items-center justify-center flex-shrink-0">
          <Bot className="w-4 h-4 text-white" />
        </div>
      )}

      {/* 消息内容 */}
      <div
        className={cn(
          "max-w-[70%] rounded-2xl px-4 py-3",
          isUser
            ? "bg-blue-500 text-white"
            : "bg-white border border-slate-200 text-slate-800"
        )}
      >
        {isUser ? (
          <p className="text-sm">{message.content}</p>
        ) : (
          <div className="prose prose-sm max-w-none">
            <ReactMarkdown
              components={{
                code({ node, inline, className, children, ...props }) {
                  const match = /language-(\w+)/.exec(className || "");
                  return !inline && match ? (
                    <SyntaxHighlighter
                      style={oneDark}
                      language={match[1]}
                      PreTag="div"
                      className="rounded-lg"
                      {...props}
                    >
                      {String(children).replace(/\n$/, "")}
                    </SyntaxHighlighter>
                  ) : (
                    <code className={className} {...props}>
                      {children}
                    </code>
                  );
                },
              }}
            >
              {message.content}
            </ReactMarkdown>
          </div>
        )}

        {/* 引用标注 */}
        {message.citations && message.citations.length > 0 && (
          <div className="mt-2 pt-2 border-t border-slate-100">
            <p className="text-xs text-slate-400">引用来源:</p>
            {message.citations.map((citation, i) => (
              <p key={i} className="text-xs text-slate-500">
                [{i + 1}] {citation}
              </p>
            ))}
          </div>
        )}
      </div>

      {/* 用户头像 */}
      {isUser && (
        <div className="w-8 h-8 rounded-full bg-slate-200 flex items-center justify-center flex-shrink-0">
          <User className="w-4 h-4 text-slate-600" />
        </div>
      )}
    </div>
  );
}
```

**特性**:
- 用户消息靠右显示，助手消息靠左显示
- 支持 Markdown 渲染
- 代码语法高亮
- 引用标注

### 3.1 ProfileRadar.tsx

**文件位置**: `src/components/profile/ProfileRadar.tsx`

**功能**: 六维雷达图，可视化学生画像

**实现方式**: 使用 SVG 自定义实现

```tsx
"use client";

import React from "react";
import type { ProfileDimensions } from "@/types";

interface ProfileRadarProps {
  dimensions: ProfileDimensions;
  size?: number;
}

export function ProfileRadar({ dimensions, size = 300 }: ProfileRadarProps) {
  const centerX = size / 2;
  const centerY = size / 2;
  const radius = size / 2 - 40;

  // 维度数据
  const维度 = [
    { key: "knowledge_mastery", label: "知识掌握", value: getAverage(dimensions.knowledge_mastery) },
    { key: "learning_style", label: "学习风格", value: getAverage(dimensions.learning_style) },
    { key: "cognitive_level", label: "认知水平", value: getAverage(dimensions.cognitive_level) },
    { key: "interest_preferences", label: "兴趣偏好", value: getAverage(dimensions.interest_preferences) },
    { key: "weak_topics", label: "薄弱环节", value: 1 - dimensions.weak_topics.length / 10 },
    { key: "learning_pace", label: "学习节奏", value: dimensions.learning_pace.daily_hours / 8 },
  ];

  // 计算多边形点
  const points = 维度.map((dim, i) => {
    const angle = (Math.PI * 2 * i) / 维度.length - Math.PI / 2;
    const x = centerX + radius * dim.value * Math.cos(angle);
    const y = centerY + radius * dim.value * Math.sin(angle);
    return `${x},${y}`;
  }).join(" ");

  return (
    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
      {/* 背景网格 */}
      {[0.25, 0.5, 0.75, 1].map((scale) => (
        <polygon
          key={scale}
          points={维度.map((_, i) => {
            const angle = (Math.PI * 2 * i) / 维度.length - Math.PI / 2;
            const x = centerX + radius * scale * Math.cos(angle);
            const y = centerY + radius * scale * Math.sin(angle);
            return `${x},${y}`;
          }).join(" ")}
          fill="none"
          stroke="#e2e8f0"
          strokeWidth="1"
        />
      ))}

      {/* 数据区域 */}
      <polygon
        points={points}
        fill="rgba(59, 130, 246, 0.2)"
        stroke="#3b82f6"
        strokeWidth="2"
      />

      {/* 数据点 */}
      {维度.map((dim, i) => {
        const angle = (Math.PI * 2 * i) / 维度.length - Math.PI / 2;
        const x = centerX + radius * dim.value * Math.cos(angle);
        const y = centerY + radius * dim.value * Math.sin(angle);
        return (
          <circle
            key={dim.key}
            cx={x}
            cy={y}
            r="4"
            fill="#3b82f6"
          />
        );
      })}

      {/* 标签 */}
      {维度.map((dim, i) => {
        const angle = (Math.PI * 2 * i) / 维度.length - Math.PI / 2;
        const x = centerX + (radius + 20) * Math.cos(angle);
        const y = centerY + (radius + 20) * Math.sin(angle);
        return (
          <text
            key={dim.key}
            x={x}
            y={y}
            textAnchor="middle"
            dominantBaseline="middle"
            className="text-xs fill-slate-600"
          >
            {dim.label}
          </text>
        );
      })}
    </svg>
  );
}

function getAverage(obj: Record<string, number>): number {
  const values = Object.values(obj);
  if (values.length === 0) return 0;
  return values.reduce((a, b) => a + b, 0) / values.length;
}
```

### 4.1 MindMapCard.tsx

**文件位置**: `src/components/mindmap/MindMapCard.tsx`

**功能**: 思维导图渲染 (Mermaid)

**重要**: 必须使用 `'use client'` + `dynamic(..., { ssr: false })`

**实现要点**:
```tsx
"use client";

import React, { useEffect, useRef, useState } from "react";
import dynamic from "next/dynamic";
import { ZoomIn, ZoomOut, Download } from "lucide-react";
import { Button } from "@/components/ui/button";

interface MindMapCardProps {
  code: string;
  className?: string;
}

export function MindMapCard({ code, className = "" }: MindMapCardProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [scale, setScale] = useState(1);

  useEffect(() => {
    renderMermaid();
  }, [code]);

  const renderMermaid = async () => {
    if (!containerRef.current || !code) return;

    setIsLoading(true);
    setError(null);

    try {
      const mermaid = (await import("mermaid")).default;

      mermaid.initialize({
        startOnLoad: false,
        theme: "default",
        securityLevel: "loose",
        mindmap: {
          padding: 20,
        },
        themeVariables: {
          primaryColor: "#3b82f6",
          primaryTextColor: "#1e293b",
          primaryBorderColor: "#93c5fd",
          lineColor: "#94a3b8",
          secondaryColor: "#e0f2fe",
          tertiaryColor: "#f0f9ff",
        },
      });

      const id = `mindmap-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;

      containerRef.current.innerHTML = "";

      const { svg } = await mermaid.render(id, code);
      containerRef.current.innerHTML = svg;

      setIsLoading(false);
    } catch (err: any) {
      console.error("Mermaid render error:", err);
      setError(err.message || "渲染失败");
      setIsLoading(false);
    }
  };

  return (
    <div className={`relative bg-white rounded-lg border border-slate-200 overflow-hidden ${className}`}>
      {/* 工具栏 */}
      <div className="absolute top-2 right-2 flex gap-1 z-10">
        <Button
          size="sm"
          variant="ghost"
          onClick={() => setScale((s) => Math.min(s + 0.2, 2))}
          className="w-7 h-7 p-0 bg-white/80 backdrop-blur"
        >
          <ZoomIn className="w-3.5 h-3.5" />
        </Button>
        <Button
          size="sm"
          variant="ghost"
          onClick={() => setScale((s) => Math.max(s - 0.2, 0.5))}
          className="w-7 h-7 p-0 bg-white/80 backdrop-blur"
        >
          <ZoomOut className="w-3.5 h-3.5" />
        </Button>
      </div>

      {/* 内容区 */}
      <div className="p-4 overflow-auto max-h-[400px]">
        {isLoading && (
          <div className="flex items-center justify-center h-32 text-sm text-slate-400">
            <div className="animate-spin w-5 h-5 border-2 border-blue-500 border-t-transparent rounded-full mr-2" />
            渲染中...
          </div>
        )}

        {error && (
          <div className="flex items-center justify-center h-32 text-sm text-red-500">
            渲染失败: {error}
          </div>
        )}

        <div
          ref={containerRef}
          className="flex justify-center"
          style={{
            transform: `scale(${scale})`,
            transformOrigin: "center top",
            display: isLoading || error ? "none" : "block",
          }}
        />
      </div>
    </div>
  );
}
```

### 5.3 ErrorBoundary.tsx

```tsx
"use client";

import React, { Component, ErrorInfo, ReactNode } from "react";
import { AlertCircle, RefreshCw } from "lucide-react";
import { Button } from "@/components/ui/button";

interface Props {
  children: ReactNode;
  fallback?: ReactNode;
}

interface State {
  hasError: boolean;
  error: Error | null;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error("ErrorBoundary caught:", error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      if (this.props.fallback) {
        return this.props.fallback;
      }

      return (
        <div className="flex flex-col items-center justify-center py-12 text-center">
          <div className="w-16 h-16 rounded-full bg-red-100 flex items-center justify-center mb-4">
            <AlertCircle className="w-8 h-8 text-red-500" />
          </div>
          <h3 className="text-lg font-medium text-slate-700 mb-2">
            出现了一些问题
          </h3>
          <p className="text-sm text-slate-500 mb-4">
            {this.state.error?.message || "未知错误"}
          </p>
          <Button
            variant="outline"
            onClick={() => this.setState({ hasError: false, error: null })}
          >
            <RefreshCw className="w-4 h-4 mr-2" />
            重试
          </Button>
        </div>
      );
    }

    return this.props.children;
  }
}
```
